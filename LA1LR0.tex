\begin{definition}[$\mathrm{LR(0)}$]
Let $G \triangleq ( N , T , P , S )$ be a CFG.
\begin{enumerate} \setlength{\itemsep}{0.2pt} \setlength{\parskip}{0.2pt}
\item[(i)] $G^\prime \triangleq ( N^\prime , T^\prime , P^\prime , S^\prime ) = ( N \uplus \set*{ S^\prime } , T \uplus \set*{ \$ } , P \cup \set*{ [ S^\prime \to S \$ ] } , S^\prime )$.
\item[(ii)] $V \triangleq N \uplus T$. $V^\prime \triangleq N^\prime \uplus T^\prime$.
\item[(iii)] $\alpha_1 \xRightarrow[\mathbf{rm}]{}_{P^\prime} \alpha_2 \xLeftrightarrow[]{\mathrm{def}} ( \alpha_1 , \alpha_2 ) \in \set*{ ( \alpha A z , \alpha \omega z ) \mid [ A \to \omega ] \in P^\prime , \alpha \in { V^\prime }^\ast , z \in { T^\prime }^\ast }$.
\item[(iv)] $L ( G ) \triangleq \set*{ z \in { T }^\ast \mid S \Rightarrow_{P}^\ast z }$. $L ( G^\prime ) \triangleq \set*{ z \in { T }^\ast \mid S^\prime \xRightarrow[\mathbf{rm}]{}_{P^\prime}^\ast z \$ }$. $L ( G ) = L ( G^\prime )$.
\item[(v)] $I_{\mathrm{LR(0)}} \triangleq \set*{ [ A \to \alpha \cdot \beta ] \mid [ A \to \alpha \beta ] \in P^\prime }$.
\item[(vi)] $\mathrm{Cl} ( q ) =_\mu q \cup \set*{ [ A \to \cdot \omega ] \mid [ A \to \omega ] \in P^\prime , [ B \to \beta \cdot A \gamma ] \in \mathrm{Cl} ( q ) }$ for $ q \in \powerset ( I_{\mathrm{LR(0)}} ) $.
\item[(vii)] $q_0 := \mathrm{Cl} ( \set*{ [ S^\prime \to \cdot S \$ ] } )$.
\item[(viii)] $\mathrm{GOTO} ( q , X ) := \mathrm{Cl} ( \set*{ [ A \to \alpha X \cdot \beta ] \mid [ A \to \alpha \cdot X \beta ] \in q } )$ for $q \in \powerset ( I_{\mathrm{LR(0)}} )$ and $X \in V^\prime$.
\item[(ix)] $Q \triangleq \mathrm{PT} \setminus \set*{ \emptyset }$ where $\mathrm{PT} =_\mu \set*{ q_0 } \cup \set*{ \mathrm{GOTO} ( q , X ) \mid q \in \mathrm{PT} , X \in V^\prime }$.
\item[(x)] $\varepsilon : p \to q \xLeftrightarrow[]{\mathrm{def}} p \in Q \land p = q$. $X \alpha : p \to q \xLeftrightarrow[]{\mathrm{def}} p \in Q \land \alpha : \mathrm{GOTO} ( p , X ) \to q$.
\item[(xi)] $\delta ( q , X ) := \mathrm{GOTO} ( q , X )$ for $q \in Q$ and $X \in V^\prime$.
\item[(xii)] $\mathrm{reduce} ( q , t ) := \set*{ [ A \to \omega ] \mid [ A \to \omega \cdot ] \in q }$ for $q \in Q$ and $t \in T^\prime$.
\item[(xiii)] $$\infer[\mathrm{Shift}]{ ( \alpha : q \to q^\prime , t z ) \vdash ( \alpha t : q \to q^{\prime \prime} , z ) }{ q \in Q & q^{\prime \prime} = \delta ( q^\prime , t ) & q^{\prime \prime} \ne \emptyset }$$
\item[(xiv)] $$\infer[\mathrm{Reduce} (A \to \omega)]{ ( \alpha \omega : q \to q^\prime , t z ) \vdash ( \alpha A : q \to q^{\prime \prime} , t z ) }{ q \in Q & [ A \to \omega ] \in \mathrm{reduce} ( q^\prime , t ) & q^{\prime \prime} = \delta ( q^\prime , A ) & q^{\prime \prime} \ne \emptyset }$$
\item[(xv)] $q_f := \delta ( \delta ( q_0 , S ) , \$ )$.
\item[(xvi)] $L ( LRA ( G ) ) \triangleq \set*{ z \in { T }^\ast \mid ( \varepsilon : q_0 \to q_0 , z \$ ) \vdash^\ast ( S \$ : q_0 \to q_f , \varepsilon ) }$. $L ( G ) = L ( LRA ( G ) )$.
\end{enumerate}
\end{definition}

\begin{theorem}[Lookahead]
Define $\mathbf{LA} : \set*{ ( q , [ A \to \omega ] ) \mid q \in Q , [ A \to \omega \cdot ] \in q } \to \powerset ( T^\prime )$ by
\begin{align}
\mathbf{LA} ( q , [ A \to \omega ] ) \triangleq \set*{ t \in T^\prime \mid S^\prime \xRightarrow[\mathbf{rm}]{}_{P^\prime}^\ast \alpha A t z , \alpha \omega : q_0 \to q , \alpha \in { V^\prime }^\ast , z \in { T^\prime }^\ast } .
\end{align}
Then, overriding $\mathrm{reduce} : Q \times T^\prime \to \powerset ( P^\prime )$ of the above $\mathrm{LR(0)}$ parser with
\begin{align*}
( q , t ) \mapsto \set*{ [ A \to \omega ] \mid [ A \to \omega \cdot ] \in q , t \in \mathbf{LA} ( q , [ A \to \omega ] ) }
\end{align*}
yields an $\mathrm{LALR(1)}$ parser if there are no conflicts among actions.
\end{theorem}

\begin{theorem}[Construction]
Letting $ M \mathbin{!} k := \set*{ v \mid ( k , v ) \in M } $, the identity
\begin{align}
\mathbf{LA} ( q , [ A \to \omega ] ) = \set*{ t \in T^\prime \mid p \in Q , \omega : p \to q ,  \delta ( p , A ) \ne \emptyset , t \in \mathbf{Follow} \mathbin{!} ( p , A ) }
\end{align}
holds whenever $q \in Q \land [ A \to \omega \cdot ] \in q$,
where $$ \mathbf{Read} \subseteq \set*{ ( ( p , A ) , t ) \mid p \in Q , A \in N^\prime , t \in T^\prime , \delta ( p , A ) \ne \emptyset } $$ and $$ \mathbf{Follow} \subseteq \set*{ ( ( p , A ) , t ) \mid p \in Q , A \in N^\prime , t \in T^\prime , \delta ( p , A ) \ne \emptyset } $$ are inductively defined by:
\[
\begin{matrix}
\infer[\mathrm{DR}]{ t \in \mathbf{Read} \mathbin{!} ( p , A ) }{ \delta ( \delta ( p , A ) , t ) \ne \emptyset } & \infer[\mathrm{reads}]{ \mathbf{Read} \mathbin{!} ( r , C ) \subseteq \mathbf{Read} \mathbin{!} ( p , A ) }{ \delta ( p , A ) = r & C \Rightarrow_{P^\prime}^\ast \varepsilon } \\
\\
\infer[\mathrm{Read}]{ t \in \mathbf{Follow} \mathbin{!} ( p , A ) }{ t \in \mathbf{Read} \mathbin{!} ( p , A ) } & \infer[\mathrm{includes}]{ \mathbf{Follow} \mathbin{!} ( p^\prime , B ) \subseteq \mathbf{Follow} \mathbin{!} ( p , A ) }{ [ B \to \beta \cdot A \gamma ] \in p & \gamma \Rightarrow_{P^\prime}^\ast \varepsilon & \beta : p^\prime \to p }
\end{matrix}
\]
\end{theorem}
