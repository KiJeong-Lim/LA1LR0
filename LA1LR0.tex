\begin{abstract}
\noindent
This note fixes the LR(0) characteristic automaton, defines the one-symbol
lookahead set attached to each completed LR(0) item, and derives the usual
Read/Follow digraph computation.  The proofs are written so that the essential
claims appear as inclusions, fixed-point equations, and derivation chains.
\end{abstract}

\section{The LR(0) Automaton}

\begin{definition}[$\LRA(G)$]\label{def:lra}
Let \(G \triangleq (N,T,P,S)\) be a context-free grammar.  Define:
\begin{enumerate}[label=\textup{(\roman*)}]
\item
\[
G^\prime=(N^\prime,T^\prime,P^\prime,S^\prime)
\triangleq
\bigl(N\uplus\set*{S^\prime},\,
      T\uplus\set*{\$},\,
      P\cup\set*{\Item{S^\prime}{S\$}},\,
      S^\prime\bigr).
\]

\item
\[
V\triangleq N\uplus T,\qquad
V^\prime\triangleq N^\prime\uplus T^\prime .
\]

\item
\[
\beta\RMP\gamma
\quad\xLeftrightarrow{\mathrm{def}}\quad
(\beta,\gamma)\in
\set*{(\alpha A z,\alpha\omega z)
\mid
\Item{A}{\omega}\in P^\prime,\ \alpha\in V^{\prime *},\ z\in T^{\prime *}}.
\]

\item
\[
I_{\LRzero}
\triangleq
\set*{\Item{A}{\alpha\cdot\beta}
\mid
\Item{A}{\alpha\beta}\in P^\prime}.
\]

\item For \(q\subseteq I_{\LRzero}\),
\[
\mathrm{Cl}(q)
=_\mu
q\cup
\set*{\Item{A}{\eps\cdot\omega}
\mid
\Item{A}{\omega}\in P^\prime,\
\Item{B}{\beta\cdot A\gamma}\in \mathrm{Cl}(q)}.
\]

\item
\[
q_0\triangleq
\mathrm{Cl}\bigl(\set*{\Item{S^\prime}{\eps\cdot S\$}}\bigr).
\]

\item For \(q\subseteq I_{\LRzero}\) and \(X\in V^\prime\),
\[
\mathrm{GOTO}(q,X)
\triangleq
\mathrm{Cl}\bigl(
\set*{\Item{A}{\alpha X\cdot\beta}
\mid
\Item{A}{\alpha\cdot X\beta}\in q}
\bigr).
\]

\item
\[
Q\triangleq \mathrm{PT}\setminus\set*{\emptyset},
\qquad
\mathrm{PT}
=_\mu
\set*{q_0}
\cup
\set*{\mathrm{GOTO}(q,X)\mid q\in\mathrm{PT},\ X\in V^\prime}.
\]

\item The path relation is defined by structural recursion on the input word:
\[
\Path{\eps}{p}{q} \Longleftrightarrow p\in Q\land p=q,
\qquad
\Path{X\alpha}{p}{q}
\Longleftrightarrow
p\in Q\land \mathrm{GOTO}(p,X)\in Q
\land \Path{\alpha}{\mathrm{GOTO}(p,X)}{q} .
\]
Thus a path blocks exactly when the next \(\mathrm{GOTO}\)-state is empty; the
relation is deterministic, not an additional least fixed point.

\item
\[
\Config
\triangleq
\set*{(\Path{\alpha}{p}{q},z)
\mid
\alpha\in V^{\prime *},\ p,q\in Q,\ z\in T^{\prime *},\
\Path{\alpha}{p}{q}}.
\]

\item
\[
\delta(q,X)\triangleq \mathrm{GOTO}(q,X)
\qquad(q\in Q,\ X\in V^\prime).
\]

\item
\[
\Reduce(q,t)
\triangleq
\set*{\Item{A}{\omega}
\mid
\Item{A}{\omega\cdot\eps}\in q}
\qquad(q\in Q,\ t\in T^\prime).
\]

\item Let \({\vdash}\subseteq\Config\times\Config\) be generated by:
\[
\infer[\mathrm{Shift}]
  {(\Path{\alpha}{q}{q^\prime},tz)
    \vdash
   (\Path{\alpha t}{q}{q^{\prime\prime}},z)}
  {q^{\prime\prime}=\delta(q^\prime,t) & q^{\prime\prime}\in Q}
\]
\[
\infer[\mathrm{Reduce}(A\to\omega)]
  {(\Path{\alpha\omega}{q}{q^\prime},tz)
    \vdash
   (\Path{\alpha A}{q}{q^{\prime\prime}},tz)}
  {\Path{\alpha}{q}{p}
   &
   \Path{\omega}{p}{q^\prime}
   &
   \Item{A}{\omega}\in\Reduce(q^\prime,t)
   &
   q^{\prime\prime}=\delta(p,A)
   &
   q^{\prime\prime}\in Q}
\]

\item
\[
q_f\triangleq\delta(\delta(q_0,S),\$).
\]

\item The accepted language is defined as
\[
L(\LRA(G))
\triangleq
\set*{z\in T^*
\mid
(\Path{\eps}{q_0}{q_0},z\$)
\vdash^*
(\Path{S\$}{q_0}{q_f},\eps)}.
\]
\end{enumerate}
\end{definition}

For \(c=(\Path{\alpha}{p}{q},u)\in\Config\), define the unread input by
\[
\mathrm{input}(c)\triangleq u .
\]
The accepting configurations are exactly
\[
\mathsf{Accept}(c)
\Longleftrightarrow
c=(\Path{S\$}{q_0}{q_f},\eps),
\]
and an error configuration is a non-accepting run configuration with no Shift or
Reduce successor under the table being executed.  In a mechanized definition,
the displayed stack path is proof-carrying: the constructors for Shift and
Reduce above include the side conditions that their target states lie in \(Q\),
so their conclusions are again elements of \(\Config\).

For \(c=(\Path{\alpha}{p}{q},u)\in\Config\), set $\mathrm{Y}(c) \in V^{\prime *}$ as
\[
\mathrm{Y}(c)\triangleq\alpha u.
\]

\begin{lemma}[One-step yield invariant]\label{lem:lra-one-step-yield}
Each parser step reverses zero or one rightmost grammar step:
\begin{equation}\label{eq:one-step-yield}
c\vdash c'
\Longrightarrow
\begin{cases}
\mathrm{Y}(c')=\mathrm{Y}(c),
  &\text{if the step is Shift},\\[2pt]
\mathrm{Y}(c')\RMP\mathrm{Y}(c),
  &\text{if the step is Reduce}.
\end{cases}
\end{equation}
\end{lemma}

\begin{proof}
Indeed,
\[
\begin{array}{rclcl}
\mathrm{Shift}:&
\mathrm{Y}(\Path{\alpha}{q}{q^\prime},tz)
&=&\alpha tz
&=\mathrm{Y}(\Path{\alpha t}{q}{q^{\prime\prime}},z),\\[2pt]
\mathrm{Reduce}:&
\mathrm{Y}(\Path{\alpha A}{q}{q^{\prime\prime}},tz)
&=&\alpha A t z
&\RMP \alpha\omega t z
=\mathrm{Y}(\Path{\alpha\omega}{q}{q^\prime},tz).
\end{array}
\]
\end{proof}

\begin{corollary}[Multi-step yield invariant]\label{cor:lra-star-yield}
For \(c,c'\in\Config\),
\begin{equation}\label{eq:star-yield}
c\vdash^* c'
\Longrightarrow
\mathrm{Y}(c')\RMP^*\mathrm{Y}(c).
\end{equation}
\end{corollary}

\begin{proof}
Induct on the length of \(c\vdash^*c'\) and use
Lemma~\ref{lem:lra-one-step-yield} at each parser step.
\end{proof}

\begin{corollary}[LR(0) soundness]\label{cor:lra-sound}
\[
L(\LRA(G))\subseteq L(G).
\]
\end{corollary}

\begin{proof}
\[
\begin{aligned}
z\in L(\LRA(G))
&\Longrightarrow
(\Path{\eps}{q_0}{q_0},z\$)
\vdash^*
(\Path{S\$}{q_0}{q_f},\eps)\\
&\overset{\eqref{eq:star-yield}}{\Longrightarrow}
S\$\RMP^*z\$\\
&\Longrightarrow
S\DerP^*z\\
&\Longrightarrow
z\in L(G).
\end{aligned}
\]
The third implication deletes the endmarker.  Write the rightmost derivation
witnessing \(S\$\RMP^*z\$\) as
\[
S\$=\gamma_0\RMP\gamma_1\RMP\cdots\RMP\gamma_M=z\$ .
\]
Since \(S^\prime\) occurs on no right-hand side of \(P^\prime\), the only
production mentioning \(S^\prime\), namely \(\Item{S^\prime}{S\$}\), is never
applicable to a form derived from \(S\$\); hence every step uses a production of
\(P\subseteq P^\prime\).  Moreover \(\$\notin V\) occurs on no right-hand side of
\(P\), and no production of \(P\) rewrites a terminal, so the rightmost symbol
\(\$\) of \(\gamma_0\) is never touched:
\[
\gamma_k=\delta_k\$,\qquad \delta_k\in V^{*}\quad(0\le k\le M),
\qquad \delta_0=S,\quad \delta_M=z,
\qquad \delta_k\DerP\delta_{k+1}.
\]
Stripping the common trailing \(\$\) therefore yields
\(S=\delta_0\DerP\delta_1\DerP\cdots\DerP\delta_M=z\), i.e.\ \(S\DerP^*z\).
\end{proof}

\begin{fact}[Elimination of non-productive nonterminals]\label{fact:productive}
Call \(A\in N^\prime\) \emph{productive} if \(A\Rightarrow_{P^\prime}^*w\) for some
\(w\in T^{\prime *}\), and \emph{non-productive} otherwise.  Removing every
non-productive nonterminal together with all productions that mention one
preserves the generated language.
\end{fact}

\begin{proof}
Extend productivity to strings: \(\sigma\in V^{\prime *}\) is productive if
\(\sigma\Rightarrow_{P^\prime}^*x\) for some \(x\in T^{\prime *}\).  A terminal
derives only itself and a derivation of a string is the concatenation of
derivations of its symbols, so
\begin{equation}\label{eq:prod-string}
Y_1\cdots Y_m\text{ productive}
\iff
\forall i.\ Y_i\text{ productive}
\qquad(t\in T^\prime\text{ productive}).
\end{equation}

\proofpart{Fixed-point characterization}
Let \(\mathrm{Gen}\subseteq N^\prime\) be the least fixed point
\[
\mathrm{Gen}=\bigcup_{n\ge 0}\mathrm{Gen}_n,
\quad
\mathrm{Gen}_0=\emptyset,
\quad
\mathrm{Gen}_{n+1}
=\set*{A\in N^\prime\mid
\exists\,\Item{A}{X_1\cdots X_k}\in P^\prime.\
\forall i.\ X_i\in T^\prime\cup\mathrm{Gen}_n}
\]
(with \(k=0\), i.e.\ \(\Item{A}{\eps}\in P^\prime\), permitted).  Then \(A\) is
productive iff \(A\in\mathrm{Gen}\):
\[
\begin{aligned}
A\in\mathrm{Gen}_{n+1}
&\Longrightarrow
\Item{A}{X_1\cdots X_k}\in P^\prime\ \text{with}\ X_i\in T^\prime\cup\mathrm{Gen}_n\\
&\overset{\mathrm{IH}_n}{\Longrightarrow}
A\Rightarrow_{P^\prime}X_1\cdots X_k\Rightarrow_{P^\prime}^*x_1\cdots x_k\in T^{\prime *};\\[2pt]
A\Rightarrow_{P^\prime}^{\,\ell}w\in T^{\prime *}\ (\ell\ge1)
&\Longrightarrow
A\Rightarrow_{P^\prime}X_1\cdots X_k\Rightarrow_{P^\prime}^{\,\ell-1}w,\
X_i\Rightarrow_{P^\prime}^{\,\ell_i}w_i,\ \ell_i<\ell\\
&\overset{\mathrm{IH}_\ell}{\Longrightarrow}
X_i\in T^\prime\cup\mathrm{Gen}\Longrightarrow A\in\mathrm{Gen}.
\end{aligned}
\]
Thus the non-productive nonterminals are exactly \(N^\prime\setminus\mathrm{Gen}\),
a least fixed point.

\proofpart{Language preservation}
Let \(\widehat G\) delete \(N^\prime\setminus\mathrm{Gen}\) and every production
mentioning it; since terminals are productive, its productions are
\[
\widehat P
\triangleq
\set*{\Item{A}{X_1\cdots X_k}\in P^\prime\mid
A,X_1,\ldots,X_k\ \text{all productive}}
\subseteq P^\prime .
\]
As \(\widehat P\subseteq P^\prime\), \(S^\prime\Rightarrow_{\widehat P}^*z\) implies
\(S^\prime\Rightarrow_{P^\prime}^*z\).  For the converse, induct on the length
\(\ell\) of \(\sigma\Rightarrow_{P^\prime}^{\,\ell}x\in T^{\prime *}\):
\[
\sigma\Rightarrow_{P^\prime}^{\,\ell}x\in T^{\prime *}
\ \Longrightarrow\
\sigma\text{ productive}\ \land\ \sigma\Rightarrow_{\widehat P}^*x .
\]
For \(\ell=0\), \(\sigma=x\in T^{\prime *}\).  For \(\ell\ge1\), split off the
first step \(\sigma=\sigma_1C\sigma_2\), \(\Item{C}{\gamma}\in P^\prime\),
\(\sigma_1,\sigma_2\in V^{\prime *}\):
\[
\sigma=\sigma_1C\sigma_2
\Rightarrow_{P^\prime}
\sigma_1\gamma\sigma_2
\Rightarrow_{P^\prime}^{\,\ell-1}x
\ \overset{\mathrm{IH}}{\Longrightarrow}\
\sigma_1\gamma\sigma_2\ \text{productive}\ \land\
\sigma_1\gamma\sigma_2\Rightarrow_{\widehat P}^*x .
\]
By \eqref{eq:prod-string}, \(\gamma\) is productive, hence so is \(C\)
(via \(\Item{C}{\gamma}\)), so \(\Item{C}{\gamma}\in\widehat P\) and
\[
\sigma=\sigma_1C\sigma_2
\Rightarrow_{\widehat P}
\sigma_1\gamma\sigma_2
\Rightarrow_{\widehat P}^*x .
\]
Taking \(\sigma=S^\prime\) gives \(L(G^\prime)\subseteq L(\widehat G)\), so
\(L(\widehat G)=L(G^\prime)\) and every nonterminal of \(\widehat G\) is
productive.

When \(L(G)\ne\emptyset\), \(S\) is productive, hence \(S^\prime\) is productive
(\(\Item{S^\prime}{S\$}\)) and \(\widehat G\) keeps the start symbol with the same
language.  This is a language-level replacement only.  It must not be used as a
``without loss of generality'' step for statements about the LR(0) automaton
itself, because \(\mathrm{Cl}\), \(\mathrm{GOTO}\), \(Q\), and all lookahead
relations are rebuilt from the production set.  Any theorem below that needs
productivity states that hypothesis explicitly.
\end{proof}

The path relation factors through the state reached after any prefix: induction
on \(|\alpha|\) using Definition~\ref{def:lra}(ix) gives, for all
\(\alpha,\omega\in V^{\prime *}\) and \(r\in Q\),
\begin{equation}\label{eq:path-concat}
\Path{\alpha\omega}{r}{q}
\iff
\exists!\,p\in Q.\ \Path{\alpha}{r}{p}\ \land\ \Path{\omega}{p}{q},
\end{equation}
the intermediate state being the unique state obtained by following \(\alpha\)
from \(r\), since each \(\delta(\cdot,X)=\mathrm{GOTO}(\cdot,X)\) is
single-valued.

\begin{lemma}[Finite closed LR(0) states]\label{lem:state-finiteness}
The LR(0) construction satisfies the following structural invariants.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \(I_{\LRzero}\) is finite, every \(q\in Q\) is a subset of
      \(I_{\LRzero}\), and \(Q\) is finite.
\item Every \(q\in Q\) is \(\mathrm{Cl}\)-closed: if
      \(\Item{B}{\beta\cdot A\gamma}\in q\) and
      \(\Item{A}{\omega}\in P^\prime\), then
      \(\Item{A}{\eps\cdot\omega}\in q\).
\item If \(q\in Q\), \(X\in V^\prime\), and \(\delta(q,X)\ne\emptyset\), then
      \(\delta(q,X)\in Q\).  Equivalently, a nonempty target produced during the
      closure construction is one of the states of \(Q\).
\end{enumerate}
\end{lemma}

\begin{proof}
The item set \(I_{\LRzero}\) is finite because \(P^\prime\) is finite and each
production has finitely many dot positions.  The operator defining
\(\mathrm{Cl}\) only adds LR(0) items, so \(\mathrm{Cl}(q)\subseteq I_{\LRzero}\)
whenever \(q\subseteq I_{\LRzero}\).  Hence the fixed point \(\mathrm{PT}\) lives
inside the finite lattice \(\Pow(I_{\LRzero})\), and
\(Q=\mathrm{PT}\setminus\set*{\emptyset}\) is finite.  Closure-closedness is just
unfolding the least fixed point defining \(\mathrm{Cl}\), and (iii) follows from
how \(\mathrm{PT}\) is closed under all \(\mathrm{GOTO}\) successors.
\end{proof}

\begin{lemma}[Derivation surgery used below]\label{lem:derivation-surgery}
The following context-free derivation facts are used as ordinary lemmas in the
subsequent proofs.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Rightmostization for terminal targets.}  If
      \(\sigma\Rightarrow_{P^\prime}^*x\in T^{\prime *}\), then
      \(\sigma\RMP^*x\).
\item \emph{Terminal right-context insertion.}  If \(\sigma\RMP^*\tau\) and
      \(z\in T^{\prime *}\), then \(\lambda\sigma z\RMP^*\lambda\tau z\) for
      every left context \(\lambda\in V^{\prime *}\).
\item \emph{Terminal concatenation decomposition.}  If
      \(\sigma_1\sigma_2\Rightarrow_{P^\prime}^*x\in T^{\prime *}\), then
      \(x=x_1x_2\) for some \(x_1,x_2\in T^{\prime *}\) with
      \(\sigma_i\Rightarrow_{P^\prime}^*x_i\).
\item \emph{FIRST/path extraction.}  Suppose a state \(r\) contains an item with
      the dot before a symbol \(X\), and \(X\Rightarrow_{P^\prime}^*t\chi\) for
      \(t\in T^\prime\).  Then, after following the nullable prefix of the
      leftmost spine witnessing this derivation, the LR(0) automaton has a
      nonempty \(t\)-transition.  Equivalently, the terminal \(t\) obtained from
      \(\mathrm{FIRST}(X)\) produces the \(\delta(\cdot,t)\ne\emptyset\) premise
      used in Lemma~\ref{lem:follow-fixed-point}.
\end{enumerate}
\end{lemma}

\begin{proof}
Items (i)--(iii) are the standard induction arguments on derivation length,
using the fact that terminals are never rewritten.  For (iv), take a leftmost
spine from \(X\) to the first terminal \(t\).  The closure rule inserts the item
for each nonterminal encountered on this spine, while the nullable siblings
before the spine are followed by \(\mathrm{GOTO}\)-steps.  The first terminal item
therefore gives a nonempty \(t\)-successor.  In Coq these four statements are
best made explicit once and then reused rather than reproved inside the main
lookahead proof.
\end{proof}

\begin{definition}[Viable prefixes and valid items]\label{def:viable-valid}
For \(\eta\in V^{\prime *}\), say that \(\eta\) is a \emph{viable prefix} of
\(G^\prime\) when it ends at some dot position of a handle in a right-sentential
form:
\begin{equation}\label{eq:vp-def}
\eta\text{ viable}
\quad\Longleftrightarrow\quad
\exists A,\alpha,\beta,\delta,w.\
\Item{A}{\alpha\beta}\in P^\prime
\land
\eta=\delta\alpha
\land
w\in T^{\prime *}
\land
S^\prime\RMP^*\delta A w .
\end{equation}
For the same \(\eta\), define the valid LR(0) items by
\begin{equation}\label{eq:valid-def}
\Item{A}{\alpha\cdot\beta}\in\mathsf{Valid}(\eta)
\quad\Longleftrightarrow\quad
\Item{A}{\alpha\beta}\in P^\prime
\land
\exists\delta,w.\
\eta=\delta\alpha
\land
w\in T^{\prime *}
\land
S^\prime\RMP^*\delta A w .
\end{equation}
Thus validity is just the item-indexed form of viability:
\begin{equation}\label{eq:valid-vp}
\mathsf{Valid}(\eta)\ne\emptyset
\quad\Longleftrightarrow\quad
\eta\text{ is a viable prefix of }G^\prime .
\end{equation}
\end{definition}

\begin{lemma}[One-sided LR(0) path invariants]\label{lem:lra-path-invariants}
For arbitrary grammars, the characteristic automaton satisfies the directions
needed for LR(0) completeness:
\begin{align}
\Path{\eta}{q_0}{q}
&\Longrightarrow
\mathsf{Valid}(\eta)\subseteq q, \label{eq:valid-sub-state}\\[2pt]
\mathsf{Valid}(\eta)\ne\emptyset
&\Longrightarrow
\exists q\in Q.\ \Path{\eta}{q_0}{q}. \label{eq:viable-implies-path}
\end{align}
Moreover, every handle occurrence in a rightmost derivation is realized by a path
and by the corresponding completed LR(0) item:
\begin{equation}\label{eq:handle}
\begin{aligned}
& S^\prime\RMP^*\eta^\prime A y
  \RMP
  \eta^\prime\omega y,
  \qquad
  \Item{A}{\omega}\in P^\prime,
  \qquad
  y=t z\in T^{\prime *} \\
&\qquad\Longrightarrow
\exists p,q\in Q.\;
\Path{\eta^\prime}{q_0}{p}
\land
\Path{\omega}{p}{q} \\
&\qquad\qquad\land
\Item{A}{\omega\cdot\eps}\in q
\land
\Item{A}{\omega}\in\Reduce(q,t).
\end{aligned}
\end{equation}
\end{lemma}

\begin{proof}
The key point is that no productivity assumption is needed for the inclusion
\(\mathsf{Valid}(\eta)\subseteq q\).  Prove it simultaneously with the following
closure-superset statement: if a kernel contains all valid items at \(\eta\) whose
left part is nonempty, and contains the seed \(\Item{S^\prime}{\eps\cdot S\$}\)
when \(\eta=\eps\), then its closure contains every item in
\(\mathsf{Valid}(\eta)\).

For a valid item with nonempty left part, the preceding dot position is valid at
the previous prefix and is put into the predecessor state by induction; the
\(\mathrm{GOTO}\)-kernel then shifts the dot.  For a valid item
\(\Item{A}{\eps\cdot\omega}\), choose the marked occurrence of \(A\) in the
witness \(S^\prime\RMP^*\eta A w\).  If the derivation has length zero, the item
is the augmented seed.  Otherwise, the step that first introduces this marked
occurrence has the form
\[
S^\prime\RMP^*\mu C\rho
\RMP
\mu\sigma_1A\sigma_2\rho,
\qquad
\Item{C}{\sigma_1A\sigma_2}\in P^\prime,
\qquad
\eta=\mu\sigma_1,
\]
and the rightmost-occurrence invariant says that the part to the left of the
marked \(A\) is never changed after that step.  The item
\(\Item{C}{\sigma_1\cdot A\sigma_2}\) is valid at \(\eta\) by the shorter witness
\(S^\prime\RMP^*\mu C\rho\), hence is already in the kernel if
\(\sigma_1\ne\eps\), or is obtained by the induction hypothesis if
\(\sigma_1=\eps\).  One closure step then adds
\(\Item{A}{\eps\cdot\omega}\).  This proves \eqref{eq:valid-sub-state} by
induction on the path.

For \eqref{eq:viable-implies-path}, use \eqref{eq:valid-vp} and induct on
\(|\eta|\).  If \(\eta=\eps\), the seed makes \(q_0\ne\emptyset\) and
\(\Path{\eps}{q_0}{q_0}\).  If \(\eta=\eta_0X\), a valid item at \(\eta\) is either
itself a shifted item with left part ending in \(X\), or the closure argument
above traces it back to such a shifted valid item.  By the induction hypothesis
there is \(p\in Q\) with \(\Path{\eta_0}{q_0}{p}\), and
\eqref{eq:valid-sub-state} puts the predecessor item with dot before \(X\) in
\(p\).  Hence \(\mathrm{GOTO}(p,X)\ne\emptyset\), so it lies in \(Q\) by
Lemma~\ref{lem:state-finiteness}, giving the required path.

Finally, the premise of \eqref{eq:handle} makes
\(\Item{A}{\omega\cdot\eps}\) valid at \(\eta^\prime\omega\).  By
\eqref{eq:viable-implies-path}, choose \(q\in Q\) with
\(\Path{\eta^\prime\omega}{q_0}{q}\), and factor this path uniquely by
\eqref{eq:path-concat} as \(\Path{\eta^\prime}{q_0}{p}\) and
\(\Path{\omega}{p}{q}\).  Equation~\eqref{eq:valid-sub-state} puts the completed
item in \(q\), and Definition~\ref{def:lra}(xii) then gives
\(\Item{A}{\omega}\in\Reduce(q,t)\).
\end{proof}

\begin{lemma}[LR(0) path exactness under productivity]
\label{lem:lra-path-exact-productive}
Assume every nonterminal of \(G^\prime\) is productive.  Then every path state is
exactly the corresponding valid-item set:
\begin{equation}\label{eq:state-eq-valid}
\Path{\eta}{q_0}{q}
\Longrightarrow
q=\mathsf{Valid}(\eta).
\end{equation}
Consequently,
\begin{equation}\label{eq:vp}
(\exists q\in Q)\ \Path{\eta}{q_0}{q}
\Longleftrightarrow
\eta\text{ is a viable prefix of }G^\prime .
\end{equation}
Moreover, every path label is realized as the prefix of a right-sentential form:
\begin{equation}\label{eq:path-realization}
\Path{\eta}{q_0}{q}
\Longrightarrow
\exists z\in T^{\prime *}.\ S^\prime\RMP^*\eta z .
\end{equation}
\end{lemma}

\begin{proof}
The inclusion \(\mathsf{Valid}(\eta)\subseteq q\) is
\eqref{eq:valid-sub-state}.  For the reverse inclusion, first prove closure
soundness under productivity.  Suppose a closure step adds
\(\Item{A}{\eps\cdot\omega}\) from
\(\Item{B}{\beta_1\cdot A\beta_2}\in\mathsf{Valid}(\eta)\), witnessed by
\(S^\prime\RMP^*\delta B w\) and \(\eta=\delta\beta_1\).  Since every nonterminal is
productive, the suffix \(\beta_2\) derives some terminal string \(x\in T^{\prime *}\);
by Lemma~\ref{lem:derivation-surgery}, this can be done rightmost in the terminal
right context.  Hence
\[
S^\prime\RMP^*\delta B w
\RMP
\delta\beta_1A\beta_2w
\RMP^*
\delta\beta_1A xw,
\]
so \(\Item{A}{\eps\cdot\omega}\in\mathsf{Valid}(\eta)\).  Thus the closure of
valid kernel items contains no invalid item.  The \(\mathrm{GOTO}\) step preserves
validity by shifting the dot:
\[
\Item{A}{\alpha\cdot X\beta}\in\mathsf{Valid}(\eta)
\Longleftrightarrow
\Item{A}{\alpha X\cdot\beta}\in\mathsf{Valid}(\eta X).
\]
An induction on \(|\eta|\) gives \(q\subseteq\mathsf{Valid}(\eta)\) for every
\(\Path{\eta}{q_0}{q}\), and therefore \eqref{eq:state-eq-valid}.
Combining \eqref{eq:state-eq-valid}, \eqref{eq:valid-vp}, and the general
viable-implies-path direction \eqref{eq:viable-implies-path} proves
\eqref{eq:vp}.

For \eqref{eq:path-realization}, let \(\Path{\eta}{q_0}{q}\).  Since \(q\in Q\),
\(q\ne\emptyset\); by \eqref{eq:state-eq-valid}, choose
\(\Item{A}{\alpha\cdot\beta}\in\mathsf{Valid}(\eta)\), witnessed by
\(\eta=\delta\alpha\), \(S^\prime\RMP^*\delta A w\), and \(w\in T^{\prime *}\).
Productivity gives \(\beta\Rightarrow_{P^\prime}^*x\in T^{\prime *}\), and
Lemma~\ref{lem:derivation-surgery} rightmostizes this derivation in the terminal
right context:
\[
S^\prime\RMP^*\delta A w
\RMP
\delta\alpha\beta w
\RMP^*
\delta\alpha xw
=
\eta xw .
\]
Thus \(z=xw\) witnesses \eqref{eq:path-realization}.
\end{proof}


\begin{lemma}[LR(0) completeness]\label{lem:lra-complete}
\[
L(G)\subseteq L(\LRA(G)).
\]
\end{lemma}

\begin{proof}
Let \(z\in L(G)\), so \(S\DerP^*z\).  Choose a rightmost derivation in \(G^\prime\):
\begin{equation}\label{eq:rightmost-chain}
S\$=\gamma_0\RMP\gamma_1\RMP\cdots\RMP\gamma_N=z\$.
\end{equation}
Write its \(k\)-th step as
\begin{equation}\label{eq:k-step}
\gamma_k=\alpha_kA_k y_k
\RMP
\alpha_k\omega_k y_k=\gamma_{k+1},
\qquad
\Item{A_k}{\omega_k}\in P^\prime,\quad y_k=t_k z_k\in T^{\prime *}.
\end{equation}
By \eqref{eq:handle}, for each \(k\) there exist \(p_k,q_k\in Q\) such that
\begin{equation}\label{eq:k-handle}
\Path{\alpha_k}{q_0}{p_k},
\qquad
\Path{\omega_k}{p_k}{q_k},
\qquad
\Item{A_k}{\omega_k\cdot\eps}\in q_k,
\qquad
\Item{A_k}{\omega_k}\in\Reduce(q_k,t_k).
\end{equation}
Reading \eqref{eq:rightmost-chain} from right to left turns each step into one
inverse reduction.  By \eqref{eq:k-handle} and \eqref{eq:path-concat},
\(\Path{\alpha_k\omega_k}{q_0}{q_k}\), so the \(\mathrm{Reduce}(A_k\to\omega_k)\)
rule applies:
\begin{equation}\label{eq:inverse-reduce}
(\Path{\alpha_k\omega_k}{q_0}{q_k},y_k)
\vdash
(\Path{\alpha_kA_k}{q_0}{\delta(p_k,A_k)},y_k)
\qquad(k=N-1,\ldots,0).
\end{equation}
These reductions are interleaved with Shift blocks.  Since \(y_k\in T^{\prime *}\),
the symbol \(A_k\) exposed at step \(k\) is the rightmost nonterminal of
\(\gamma_k=\alpha_kA_ky_k\); using the previous step
\(\gamma_k=\alpha_{k-1}\omega_{k-1}y_{k-1}\) with \(y_{k-1}\in T^{\prime *}\) and
cancelling the common terminal suffix,
\begin{equation}\label{eq:shift-block}
\alpha_kA_k\,u_k=\alpha_{k-1}\omega_{k-1},
\qquad
y_k=u_k\,y_{k-1},
\qquad
u_k\in T^{\prime *}
\qquad(1\le k\le N-1).
\end{equation}
By \eqref{eq:path-concat} the path \(\Path{\alpha_{k-1}\omega_{k-1}}{q_0}{q_{k-1}}\)
factors through \(\delta(p_k,A_k)\) after the prefix \(\alpha_kA_k\), so the
\(|u_k|\) symbols of \(u_k\) are shifted one by one:
\[
(\Path{\alpha_kA_k}{q_0}{\delta(p_k,A_k)},\,u_ky_{k-1})
\ \vdash^{|u_k|}\
(\Path{\alpha_{k-1}\omega_{k-1}}{q_0}{q_{k-1}},\,y_{k-1}).
\]
At the bottom (\(k=N-1\)), the run starts by shifting all of
\(\alpha_{N-1}\omega_{N-1}\) out of \(z\$=\gamma_N\):
\[
(\Path{\eps}{q_0}{q_0},z\$)
\ \vdash^{|\alpha_{N-1}\omega_{N-1}|}\
(\Path{\alpha_{N-1}\omega_{N-1}}{q_0}{q_{N-1}},y_{N-1}).
\]
At the top (\(k=0\)) we have \(\gamma_0=S\$\), whence \(\alpha_0=\eps\),
\(A_0=S\), \(y_0=\$\); \eqref{eq:inverse-reduce} leaves the stack \(S\) reading
\(\$\), and one final Shift of \(\$\) gives
\[
(\Path{S}{q_0}{\delta(q_0,S)},\$)
\ \vdash\
(\Path{S\$}{q_0}{q_f},\eps),
\qquad
q_f=\delta(\delta(q_0,S),\$).
\]
Concatenating the Shift blocks with \eqref{eq:inverse-reduce} for
\(k=N-1,\ldots,0\) gives
\[
(\Path{\eps}{q_0}{q_0},z\$)
\vdash^*
(\Path{S\$}{q_0}{q_f},\eps),
\]
hence \(z\in L(\LRA(G))\).
\end{proof}

\begin{corollary}[Correctness of \(\LRA(G)\)]\label{cor:lra-correct}
\[
L(G)=L(\LRA(G)),
\qquad
L(G)\triangleq\set*{z\in T^*\mid S\DerP^*z}.
\]
\end{corollary}

\begin{proof}
Corollary~\ref{cor:lra-sound} and Lemma~\ref{lem:lra-complete} give
\[
L(\LRA(G))\subseteq L(G),
\qquad
L(G)\subseteq L(\LRA(G)).
\]
\end{proof}

\section{Read and Follow Sets}

Define
\(
\Look:
\set*{(q,\Item{A}{\omega})
\mid q\in Q,\ \Item{A}{\omega\cdot\eps}\in q}
\longrightarrow \Pow(T^\prime)
\)
by
\begin{equation}\label{eq:la-def}
\Look(q,\Item{A}{\omega})
\triangleq
\set*{t\in T^\prime
\mid
S^\prime\RMP^*\alpha A t z,\
\Path{\alpha\omega}{q_0}{q},\
\alpha\in V^{\prime *},\
z\in T^{\prime *}}.
\end{equation}

The remainder of this section computes \(\Look(q,\Item{A}{\omega})\) from the
LR(0) automaton, as the least fixed point of a Read/Follow digraph.

Let
\[
D\triangleq
\set*{(p,A)\mid p\in Q,\ A\in N^\prime,\ \delta(p,A)\ne\emptyset}.
\]
For a relation \(R\), write
\[
R\mathbin{!}x\triangleq\set*{y\mid (x,y)\in R}.
\]
Define \(\Read,\Follow\subseteq D\times T^\prime\) as the least relations
generated by the rules
\[
\begin{array}{cc}
\infer[\mathrm{DR}]
  {t\in\Read\mathbin{!}(p,A)}
  {\delta(\delta(p,A),t)\ne\emptyset}
&
\infer[\mathrm{reads}]
  {\Read\mathbin{!}(r,C)\subseteq\Read\mathbin{!}(p,A)}
  {\delta(p,A)=r & C\Rightarrow_{P^\prime}^*\eps & (r,C)\in D}
\\[4ex]
\infer[\mathrm{Read}]
  {t\in\Follow\mathbin{!}(p,A)}
  {t\in\Read\mathbin{!}(p,A)}
&
\infer[\mathrm{includes}]
  {\Follow\mathbin{!}(p^\prime,B)\subseteq\Follow\mathbin{!}(p,A)}
  {\Item{B}{\beta\cdot A\gamma}\in p
   & \Path{\beta}{p^\prime}{p}
   & \gamma\Rightarrow_{P^\prime}^*\eps
   & (p^\prime,B)\in D}
\end{array}
\]

Define the semantic read and follow sets on \(D\):
\begin{align}
\mathsf{Read}(p,A)
&\triangleq
\set*{t\in T^\prime
\mid
\exists r,s,\gamma.\ 
r=\delta(p,A)
\land
\gamma\Rightarrow_{P^\prime}^*\eps
\land
\gamma t:r\to s}, \label{eq:sem-read}\\
\mathsf{Follow}(p,A)
&\triangleq
\set*{t\in T^\prime
\mid
\exists\alpha,z.\ 
S^\prime\RMP^*\alpha A t z
\land
\Path{\alpha}{q_0}{p}
\land
z\in T^{\prime *}}. \label{eq:sem-follow}
\end{align}

For \((p,A)\in D\), set
\[
\mathrm{DR}(p,A)
\triangleq
\set*{t\in T^\prime\mid \delta(\delta(p,A),t)\ne\emptyset}.
\]
Let
\[
\Phi_R(F)(p,A)
=
\mathrm{DR}(p,A)
\cup
\bigcup_{\substack{\delta(p,A)=r\\ C\Rightarrow_{P^\prime}^*\eps\\(r,C)\in D}}
F(r,C).
\]

\begin{lemma}[Semantic Read fixed point]\label{lem:read-fixed-point}
For \((p,A)\in D\),
\begin{equation}\label{eq:read-eq}
\Read\mathbin{!}(p,A)
=
\mu\Phi_R(p,A)
=
\mathsf{Read}(p,A).
\end{equation}
\end{lemma}

\begin{proof}
The two \(\Read\)-rules say exactly that \(\Read\mathbin{!}\) is the least
\(\Phi_R\)-closed family, i.e.\ \(\Read\mathbin{!}(p,A)=\mu\Phi_R(p,A)\); since
\(\Phi_R\) is monotone on the finite lattice \(D\to\Pow(T^\prime)\),
\[
\mu\Phi_R=\bigcup_{m\ge 0}\Phi_R^{m}(\emptyset).
\]
Write \((p,A)\rightsquigarrow_R(r,C)\) for the reads edge
\(r=\delta(p,A)\land C\Rightarrow_{P^\prime}^*\eps\land(r,C)\in D\).  Because
\(\Phi_R(\emptyset)(p,A)=\mathrm{DR}(p,A)\), an induction on \(m\) unfolds the
operator into reads-paths:
\begin{equation}\label{eq:read-unfold}
\Phi_R^{m+1}(\emptyset)(p,A)
=\bigcup_{i=0}^{m}\
\bigcup_{(p,A)\rightsquigarrow_R^{\,i}(r,C)}
\mathrm{DR}(r,C).
\end{equation}
By \eqref{eq:sem-read}, \(t\in\mathsf{Read}(p,A)\) iff for some nullable
\(\gamma=C_1\cdots C_n\in N^{\prime *}\),
\[
r_0=\delta(p,A),\qquad
r_i=\delta(r_{i-1},C_i)\ (1\le i\le n),\qquad
\delta(r_n,t)\ne\emptyset ,
\]
If \(n=0\), this says exactly that \(t\in\mathrm{DR}(p,A)\).  If \(n>0\), it is
an \(n\)-edge reads-path ending at a \(\mathrm{DR}\)-seed holding \(t\):
\[
(p,A)\rightsquigarrow_R(r_0,C_1)\rightsquigarrow_R\cdots\rightsquigarrow_R
(r_{n-1},C_n),
\qquad
t\in\mathrm{DR}(r_{n-1},C_n)
\quad\bigl(\delta(\delta(r_{n-1},C_n),t)=\delta(r_n,t)\ne\emptyset\bigr).
\]
In both cases \eqref{eq:read-unfold} puts
\(t\) in \(\mu\Phi_R(p,A)\), and every member of the union
\eqref{eq:read-unfold} reads back as such a nullable \(\delta\)-chain.  Hence
\eqref{eq:read-eq}.
\end{proof}

Let
\[
\Phi_F(F)(p,A)
=
\mathsf{Read}(p,A)
\cup
\bigcup_{\substack{
  \Item{B}{\beta\cdot A\gamma}\in p\\
  \Path{\beta}{p^\prime}{p}\\
  \gamma\Rightarrow_{P^\prime}^*\eps\\
  (p^\prime,B)\in D}}
F(p^\prime,B).
\]

\begin{lemma}[General Follow over-approximation]\label{lem:follow-overapprox}
For arbitrary grammars and every \((p,A)\in D\),
\begin{equation}\label{eq:follow-overapprox}
\mathsf{Follow}(p,A)
\subseteq
\mu\Phi_F(p,A)
=
\Follow\mathbin{!}(p,A).
\end{equation}
\end{lemma}

\begin{proof}
By Lemma~\ref{lem:read-fixed-point}, the two \(\Follow\)-rules define the least
fixed point of \(\Phi_F\).  We prove
\(\mathsf{Follow}(p,A)\subseteq\mu\Phi_F(p,A)\) by induction on a marked
rightmost derivation witnessing
\[
S^\prime\RMP^*\alpha A t z,
\qquad
\Path{\alpha}{q_0}{p}.
\]
The occurrence of \(A\) cannot be the initial \(S^\prime\) occurrence, because it
is followed by the terminal \(t\).  Let the step that first introduces this
marked occurrence be
\[
S^\prime\RMP^*
\alpha^\prime B t^\prime z^\prime
\RMP
\alpha^\prime\beta A\gamma t^\prime z^\prime
\RMP^*
\alpha A t z,
\qquad
\alpha=\alpha^\prime\beta,
\qquad
\Item{B}{\beta A\gamma}\in P^\prime .
\]
The marked-occurrence invariant for rightmost derivations says that, after this
introducing step, the left context \(\alpha^\prime\beta\) and the marked \(A\)
are never changed.  Hence all remaining rewrites occur in the suffix descended
from \(\gamma t^\prime z^\prime\).  Also
\(\Item{B}{\beta\cdot A\gamma}\in\mathsf{Valid}(\alpha)\); since
\(\Path{\alpha}{q_0}{p}\), the one-sided invariant
\eqref{eq:valid-sub-state} gives
\begin{equation}\label{eq:includes-item-in-state}
\Item{B}{\beta\cdot A\gamma}\in p .
\end{equation}

Now split by the image of this particular \(\gamma\)-block in the displayed
suffix derivation, not by the mere nullability of \(\gamma\).

If the image of \(\gamma\) is nonempty, let \(t\) be its first terminal in the
final suffix.  Then
\(\gamma=\gamma_0X\gamma_1\) where
\(\gamma_0\Rightarrow_{P^\prime}^*\eps\) and
\(X\Rightarrow_{P^\prime}^*t\chi\).  Starting from
\eqref{eq:includes-item-in-state}, the state \(\delta(p,A)\) contains the item
\(\Item{B}{\beta A\cdot\gamma}\); following the nullable prefix \(\gamma_0\) and
using the FIRST/path extraction lemma gives a path
\(\Path{\gamma_0 t}{\delta(p,A)}{s}\) for some \(s\).  Thus
\(t\in\mathsf{Read}(p,A)\), and Lemma~\ref{lem:read-fixed-point} puts \(t\) in the
Read seed of \(\Phi_F(p,A)\), hence in \(\mu\Phi_F(p,A)\).

If the image of \(\gamma\) is empty, then
\(\gamma\Rightarrow_{P^\prime}^*\eps\) and the first terminal following the marked
\(A\) is the old \(t^\prime\), so \(t=t^\prime\).  Factor the path
\(\Path{\alpha^\prime\beta}{q_0}{p}\) uniquely as
\(\Path{\alpha^\prime}{q_0}{p^\prime}\) and
\(\Path{\beta}{p^\prime}{p}\).  The witness
\(S^\prime\RMP^*\alpha^\prime B t z\) is strictly shorter than the original
marked witness, so the induction hypothesis gives
\(t\in\mu\Phi_F(p^\prime,B)\).

It remains to check that the includes edge is genuinely present.  We already
have \eqref{eq:includes-item-in-state}, \(\Path{\beta}{p^\prime}{p}\), and
\(\gamma\Rightarrow_{P^\prime}^*\eps\).  The fourth premise
\((p^\prime,B)\in D\) follows from the same marked-occurrence argument.  If
\(B=S^\prime\), then the only \(S^\prime\)-production is
\(S^\prime\to S\$\), and the suffix \(\gamma\) in an item
\(S^\prime\to\beta A\gamma\) with the dot before \(A\) contains \(\$\), hence is
not nullable.  Thus \(B\ne S^\prime\).  The live occurrence of \(B\) in
\(S^\prime\RMP^*\alpha^\prime B t z\) was therefore introduced by some production
with the dot before \(B\); that item is valid at \(\alpha^\prime\), and
\eqref{eq:valid-sub-state} puts it in \(p^\prime\).  Hence
\(\delta(p^\prime,B)\ne\emptyset\), i.e. \((p^\prime,B)\in D\).  The includes edge
propagates \(t\) from \((p^\prime,B)\) to \((p,A)\), so
\(t\in\mu\Phi_F(p,A)\).
\end{proof}

\begin{lemma}[Semantic Follow fixed point under productivity]\label{lem:follow-fixed-point}
Assume every nonterminal of \(G^\prime\) is productive.  For \((p,A)\in D\),
\begin{equation}\label{eq:follow-eq}
\Follow\mathbin{!}(p,A)
=
\mu\Phi_F(p,A)
=
\mathsf{Follow}(p,A).
\end{equation}
\end{lemma}

\begin{proof}
The inclusion \(\mathsf{Follow}(p,A)\subseteq\mu\Phi_F(p,A)\) is
Lemma~\ref{lem:follow-overapprox}.  It remains to show that
\(\mathsf{Follow}\) is closed under \(\Phi_F\), for then the least fixed point is
contained in \(\mathsf{Follow}\).

For a \(\mathsf{Read}\)-seed, fix \(t\in\mathsf{Read}(p,A)\).  Since
\((p,A)\in D\), the state \(p\) is reachable from \(q_0\); choose
\(\alpha\) with \(\Path{\alpha}{q_0}{p}\).  By \eqref{eq:sem-read}, for some
nullable \(\gamma\) there is a path
\(\Path{\gamma t}{\delta(p,A)}{s}\), hence
\(\Path{\alpha A\gamma t}{q_0}{s}\).  Productivity and
Lemma~\ref{lem:lra-path-exact-productive} realize this path as a prefix of a
right-sentential form: there exists \(z\in T^{\prime *}\) such that
\[
S^\prime\RMP^*\alpha A\gamma t z .
\]
By Lemma~\ref{lem:derivation-surgery}, the nullable \(\gamma\) may be erased by
rightmost steps in the terminal right context \(t z\), giving
\[
S^\prime\RMP^*\alpha A\gamma t z
\RMP^*
\alpha A t z .
\]
Thus \(t\in\mathsf{Follow}(p,A)\).

For an includes edge, assume
\[
\Item{B}{\beta\cdot A\gamma}\in p,
\qquad
\Path{\beta}{p^\prime}{p},
\qquad
\gamma\Rightarrow_{P^\prime}^*\eps,
\qquad
(p^\prime,B)\in D,
\]
and take \(t\in\mathsf{Follow}(p^\prime,B)\), witnessed by
\(S^\prime\RMP^*\alpha^\prime B t z\) and
\(\Path{\alpha^\prime}{q_0}{p^\prime}\).  The item supplies the production
\(B\to\beta A\gamma\), so
\[
S^\prime\RMP^*\alpha^\prime B t z
\RMP
\alpha^\prime\beta A\gamma t z
\RMP^*
\alpha^\prime\beta A t z,
\]
where the last segment erases \(\gamma\) in the terminal right context.  By
\eqref{eq:path-concat}, \(\Path{\alpha^\prime\beta}{q_0}{p}\), hence
\(t\in\mathsf{Follow}(p,A)\).  Therefore
\(\mathsf{Follow}\) is \(\Phi_F\)-closed, and the least fixed point equality
follows.
\end{proof}


\begin{remark}[Marked occurrences for a Coq formalization]
The informal phrase ``the occurrence introduced by this production'' is used in
three places: the closure-superset direction of
Lemma~\ref{lem:lra-path-invariants}, the converse/over-approximation direction
of Lemma~\ref{lem:follow-overapprox}, and the proof that the includes edge really
has \((p^\prime,B)\in D\).  In a formal development this should be represented
once, either as a marked derivation relation or as a zipper over sentential
forms.

The reusable invariant is simple: in a rightmost derivation, once a marked
occurrence survives a step, the string strictly to its left is never changed by
later steps.  With this invariant, the introducing step for a marked \(A\) has a
well-scoped factorization
\[
S^\prime\RMP^*\alpha^\prime B t^\prime z^\prime
\RMP
\alpha^\prime\beta A\gamma t^\prime z^\prime
\RMP^*
\alpha^\prime\beta A t z,
\]
and the two cases in Lemma~\ref{lem:follow-overapprox} are determined by the
actual image of the marked \(\gamma\)-block in this derivation segment.  If that
image is nonempty, the first terminal comes from a FIRST/path lemma and gives a
Read seed.  If that image is empty, then \(\gamma\Rightarrow^*\eps\), the terminal
lookahead is inherited from the shorter \(B\)-witness, and the includes edge is
available; the separate \((p^\prime,B)\in D\) premise follows by applying the same
marked-occurrence invariant to the live occurrence of \(B\).  Thus the case split
is not ``\(\gamma\) nullable or not'', but ``the tracked \(\gamma\)-occurrence has
empty image or nonempty image in this particular derivation''.
\end{remark}


\begin{definition}[Digraph lookahead]\label{def:digraph-lookahead}
For every completed item \(\Item{A}{\omega\cdot\eps}\in q\), define the lookahead
set computed from the Read/Follow graph by
\begin{equation}\label{eq:la-follow-formula}
\widehat{\Look}(q,\Item{A}{\omega})
\triangleq
\set*{t\in T^\prime
\mid
\exists p\in Q.\
\Path{\omega}{p}{q}
\land
\delta(p,A)\ne\emptyset
\land
 t\in\Follow\mathbin{!}(p,A)}.
\end{equation}
\end{definition}

\begin{corollary}[Digraph lookahead is a safe over-approximation]
\label{cor:lookahead-overapprox}
For arbitrary grammars, whenever \(q\in Q\) and
\(\Item{A}{\omega\cdot\eps}\in q\),
\begin{equation}\label{eq:lookahead-overapprox}
\Look(q,\Item{A}{\omega})
\subseteq
\widehat{\Look}(q,\Item{A}{\omega}).
\end{equation}
\end{corollary}

\begin{proof}
Take \(t\in\Look(q,\Item{A}{\omega})\), witnessed by
\[
S^\prime\RMP^*\alpha A t z,
\qquad
\Path{\alpha\omega}{q_0}{q}.
\]
By \eqref{eq:path-concat}, factor the path as
\(\Path{\alpha}{q_0}{p}\) and \(\Path{\omega}{p}{q}\).  Then
\(t\in\mathsf{Follow}(p,A)\) by \eqref{eq:sem-follow}.  If \(A=S^\prime\), this
situation cannot occur because \(S^\prime\) occurs on no right-hand side and the
initial occurrence is not followed by a terminal in any form derived from
\(S^\prime\).  Hence \(A\ne S^\prime\).  Applying the marked-occurrence invariant
to the live occurrence of \(A\) in \(S^\prime\RMP^*\alpha A t z\) yields a valid
item at \(\alpha\) with the dot before \(A\); by \eqref{eq:valid-sub-state} that
item lies in \(p\).  Thus \(\delta(p,A)\ne\emptyset\), so \((p,A)\in D\).  Finally,
Lemma~\ref{lem:follow-overapprox} gives
\(t\in\Follow\mathbin{!}(p,A)\), and the displayed definition of
\(\widehat{\Look}\) applies.
\end{proof}

\begin{corollary}[Lookahead by Follow under productivity]\label{cor:lookahead-follow}
Assume every nonterminal of \(G^\prime\) is productive.  Whenever \(q\in Q\) and
\(\Item{A}{\omega\cdot\eps}\in q\),
\begin{equation}\label{eq:lookahead-exact}
\Look(q,\Item{A}{\omega})
=
\widehat{\Look}(q,\Item{A}{\omega}).
\end{equation}
\end{corollary}

\begin{proof}
The inclusion \(\Look\subseteq\widehat{\Look}\) is
Corollary~\ref{cor:lookahead-overapprox}.  Conversely, suppose
\(t\in\widehat{\Look}(q,\Item{A}{\omega})\), witnessed by \(p\) with
\(\Path{\omega}{p}{q}\), \(\delta(p,A)\ne\emptyset\), and
\(t\in\Follow\mathbin{!}(p,A)\).  By Lemma~\ref{lem:follow-fixed-point},
\(t\in\mathsf{Follow}(p,A)\); choose witnesses
\(S^\prime\RMP^*\alpha A t z\) and \(\Path{\alpha}{q_0}{p}\).  Then
\eqref{eq:path-concat} gives \(\Path{\alpha\omega}{q_0}{q}\), and the semantic
definition \eqref{eq:la-def} gives \(t\in\Look(q,\Item{A}{\omega})\).
\end{proof}

\begin{theorem}[Digraph computation of lookahead]\label{thm:read-follow}
For arbitrary grammars, the Read/Follow digraph computes
\(\widehat{\Look}\), a safe over-approximation of the semantic lookahead
\(\Look\).  If every nonterminal of \(G^\prime\) is productive, then
\(\widehat{\Look}=\Look\) for every completed item.
\end{theorem}

\begin{proof}
The Read fixed point is Lemma~\ref{lem:read-fixed-point}; the general semantic
inclusion for Follow is Lemma~\ref{lem:follow-overapprox}, which gives
Corollary~\ref{cor:lookahead-overapprox}.  Under the explicit productivity
hypothesis, Lemma~\ref{lem:follow-fixed-point} upgrades the inclusion to equality,
and Corollary~\ref{cor:lookahead-follow} gives the exact semantic formula.
\end{proof}


\section{Lookahead Restriction}

Recall the semantic lookahead set \(\Look\) of \eqref{eq:la-def}.  The same
language-preservation argument below applies to any guard \(K\) satisfying
\[
\Look(q,\Item{A}{\omega})\subseteq K(q,\Item{A}{\omega})
\subseteq T^\prime
\]
for completed items, because soundness only uses
\(\Reduce_K\subseteq\Reduce\) and completeness only uses
\(\Look\subseteq K\).  In particular, for arbitrary grammars one may take the
digraph-computed over-approximation \(K=\widehat{\Look}\) from
Corollary~\ref{cor:lookahead-overapprox}; under productivity this is the exact
semantic lookahead by Corollary~\ref{cor:lookahead-follow}.  We state the formulas
with the exact semantic guard \(\Look\).  Replace the LR(0)
reduction function by
\begin{equation}\label{eq:la-reduce}
\Reduce_{\Look}(q,t)
\triangleq
\set*{\Item{A}{\omega}
\mid
\Item{A}{\omega\cdot\eps}\in q,\
t\in\Look(q,\Item{A}{\omega})}.
\end{equation}

Let \(\vdash_{\Look}\) be the transition relation obtained from
\(\vdash\) by replacing \(\Reduce\) with \(\Reduce_{\Look}\), and let
\(L(\vdash_{\Look})\) be its accepted language.

\begin{lemma}[Guarded soundness]\label{lem:lookahead-sound}
\[
L(\vdash_{\Look})\subseteq L(G).
\]
\end{lemma}

\begin{proof}
From~\eqref{eq:la-reduce},
\[
\Reduce_{\Look}(q,t)\subseteq\Reduce(q,t)
\quad\Longrightarrow\quad
{\vdash_{\Look}}\subseteq{\vdash}
\quad\Longrightarrow\quad
{\vdash_{\Look}}^{\,*}\subseteq{\vdash}^*.
\]
Therefore
\begin{equation}\label{eq:la-sound}
L(\vdash_{\Look})
\subseteq
L(\LRA(G))
=
L(G)
\qquad\text{by Corollary~\ref{cor:lra-correct}.}
\end{equation}
\end{proof}

\begin{lemma}[Guarded completeness]\label{lem:lookahead-complete}
\[
L(G)\subseteq L(\vdash_{\Look}).
\]
\end{lemma}

\begin{proof}
Let \(z\in L(G)\).  In the accepting computation constructed in
Lemma~\ref{lem:lra-complete}, every Reduce step has the form
\[
(\Path{\alpha\omega}{q_0}{q},t z^\prime)
\vdash
(\Path{\alpha A}{q_0}{\delta(p,A)},t z^\prime),
\]
where
\[
S^\prime\RMP S\$\RMP^*\alpha A t z^\prime,
\qquad
\Path{\alpha\omega}{q_0}{q},
\qquad
\Item{A}{\omega\cdot\eps}\in q.
\]
By \eqref{eq:la-def},
\[
t\in\Look(q,\Item{A}{\omega})
\quad\Longleftrightarrow\quad
\Item{A}{\omega}\in\Reduce_{\Look}(q,t).
\]
Thus every Reduce step used in that accepting LR(0) computation is also legal
for \(\vdash_{\Look}\); Shift steps are unchanged.  Hence
\[
L(G)\subseteq L(\vdash_{\Look}).
\]
\end{proof}

\begin{corollary}[Language preservation]\label{cor:lookahead-language}
\[
L(\vdash_{\Look})=L(G).
\]
\end{corollary}

\begin{proof}
Combine Lemmas~\ref{lem:lookahead-sound} and~\ref{lem:lookahead-complete}.
\end{proof}


\begin{definition}[\texorpdfstring{\(\LALRone\)}{LALR(1)} action table]\label{def:lalr-table}
For a chosen one-symbol guard \(K\) on completed items, the ACTION entries are
\[
\mathsf{Act}_K(q,t)
\triangleq
\set*{\mathsf{shift}(\delta(q,t))\mid \delta(q,t)\in Q}
\cup
\set*{\mathsf{reduce}(A\to\omega)
\mid
\Item{A}{\omega\cdot\eps}\in q
\land
t\in K(q,\Item{A}{\omega})}.
\]
A guarded LR(0) table is an \(\LALRone\) table when
\[
\forall q\in Q,\ t\in T^\prime.\quad |\mathsf{Act}_K(q,t)|\le 1 .
\]
The associated parser is the transition system obtained by executing this
single ACTION entry when it exists; if no ACTION entry exists and the
configuration is not accepting, the configuration is an error configuration.
\end{definition}

\begin{lemma}[Conflict-free table condition]\label{lem:lookahead-table}
If the resulting table has no shift/reduce or reduce/reduce conflict, its
action is single-valued under one-symbol lookahead, hence it is an
\(\LALRone\) table.
\end{lemma}

\begin{proof}
At \((q,t)\), the possible actions are
\[
\delta(q,t)\quad\text{when this shift target is nonempty},
\qquad
\Reduce_{\Look}(q,t).
\]
Absence of shift/reduce and reduce/reduce conflicts means that this action is
single-valued under one-symbol lookahead, which is precisely Definition~\ref{def:lalr-table}.
\end{proof}

\begin{theorem}[Lookahead-guarded reductions]\label{thm:lookahead-guard}
If the guarded table is conflict-free, it is an \(\LALRone\) parser accepting
\(L(G)\).
\end{theorem}

\begin{proof}
Corollary~\ref{cor:lookahead-language} gives acceptance of \(L(G)\), and
Lemma~\ref{lem:lookahead-table} gives the \(\LALRone\) table condition under the
stated absence of conflicts.
\end{proof}

\appendix
\section{Digraph Algorithm and Implementation Specification}

This appendix gives an executable specification for computing the sets used in
Theorem~\ref{thm:read-follow}.  The construction is the usual digraph view of
\(\LALRone\) lookahead computation: direct seed sets are placed on nodes,
dependency edges describe set inclusions, and the least fixed point is obtained
by collapsing strongly connected components and propagating set unions along the
resulting DAG
\cite{DeRemerPennello1982,Tarjan1972,Gallier2008}.

\subsection{General Digraph Fixed-Point Problem}

Let \(X\) be a finite set of nodes, let \(E\subseteq X\times X\) be a directed
edge relation, and let \(\mathrm{Seed}:X\to\Pow(T^\prime)\).  We use the edge
orientation
\[
x\leadsto y
\qquad\text{to mean}\qquad
F(y)\subseteq F(x).
\]
Equivalently, define the monotone operator
\[
\Psi(F)(x)
\triangleq
\mathrm{Seed}(x)\cup\bigcup_{x\leadsto y}F(y).
\]
The desired solution is the least fixed point
\begin{equation}\label{eq:digraph}
F=\mu\Psi,
\qquad
F(x)=\mathrm{Seed}(x)\cup\bigcup_{x\leadsto y}F(y).
\end{equation}

\paragraph{Verified algorithm: work-list Kleene iteration.}
For formal verification, take \(\mathrm{DIGRAPH}(X,E,\mathrm{Seed})\) to be the
following finite work-list computation.
\begin{enumerate}
\item Initialize \(F(x)\leftarrow\mathrm{Seed}(x)\) for all \(x\in X\), and put all
      edges \(x\leadsto y\) in a work-list.
\item While the work-list is nonempty, remove an edge \(x\leadsto y\).  If
      \(F(y)\nsubseteq F(x)\), replace
      \[
      F(x)\leftarrow F(x)\cup F(y)
      \]
      and reinsert every incoming edge \(u\leadsto x\).
\item Return \(F\) when the work-list is empty.
\end{enumerate}
This is the implementation target used by the specification.  It terminates
because every update strictly increases some finite set \(F(x)\subseteq T^\prime\),
so at most \(|X|\cdot |T^\prime|\) element insertions occur.  The loop invariant is
\(\mathrm{Seed}(x)\subseteq F(x)\subseteq\mu\Psi(x)\) for all \(x\).  When the
work-list is empty, every edge satisfies \(F(y)\subseteq F(x)\), hence
\(\Psi(F)\subseteq F\); together with \(\mathrm{Seed}\subseteq F\), this says
\(F\) is \(\Psi\)-closed.  Since \(\mu\Psi\) is the least \(\Psi\)-closed family
and the invariant gives \(F\subseteq\mu\Psi\), the returned value is exactly
\(\mu\Psi\).

\paragraph{SCC optimization.}
The usual Tarjan-SCC/condensation-DAG algorithm computes the same fixed point by
collapsing cycles and processing the resulting DAG in reverse topological order.
It is a valid optimization once accompanied by a separate refinement proof that
its result equals the work-list fixed point above.  The mathematical development
here relies only on the work-list correctness theorem, not on the internal
correctness of a particular SCC implementation.

\paragraph{Complexity.}
Let \(b=\lceil |T^\prime|/w\rceil\), where \(w\) is the machine word size used
for bitsets.  The work-list implementation performs at most
\(|X|\cdot |T^\prime|\) element insertions and scans incident edges when a set
grows.  With bitsets this is bounded by the usual
\[
O\bigl((|X|+|E|)\,|T^\prime|\,b\bigr)
\]
worst-case work-list bound; the SCC implementation improves the practical bound
to \(O((|X|+|E|)b)\) after the SCC refinement proof is supplied.

\subsection{\texorpdfstring{Instance for \(\Read\)}{Instance for Read}}

First compute nullable nonterminals by the least fixed point
\[
\mathrm{Null}(A)
\Longleftrightarrow
\exists \Item{A}{X_1\cdots X_k}\in P^\prime
\text{ such that every }X_i\text{ is nullable},
\]
where terminals are never nullable and \(k=0\) is allowed.  Extend this to
strings by
\[
\mathrm{Null}(X_1\cdots X_k)
\Longleftrightarrow
\bigwedge_{i=1}^k \mathrm{Null}(X_i).
\]

\begin{lemma}[Null correctness]\label{lem:null-correct}
For every \(\gamma\in V^{\prime *}\),
\[
\mathrm{Null}(\gamma)
\Longleftrightarrow
\gamma\Rightarrow_{P^\prime}^*\eps .
\]
In particular, for \(A\in N^\prime\), \(\mathrm{Null}(A)\) iff
\(A\Rightarrow_{P^\prime}^*\eps\).
\end{lemma}

\begin{proof}
The forward direction is induction on the stage at which a nonterminal enters
the least fixed point, followed by concatenating the derivations of the symbols
of \(\gamma\).  The reverse direction is induction on the length of a derivation
to \(\eps\).  A terminal cannot occur in such a derivation; for a first step
\(A\Rightarrow X_1\cdots X_k\Rightarrow^*\eps\), terminal concatenation
decomposition gives \(X_i\Rightarrow^*\eps\) for each \(i\), so the induction
hypothesis makes every \(X_i\) nullable and the fixed-point rule adds \(A\).  The
string case follows componentwise.
\end{proof}

For \(x=(p,A)\in D\), define the direct-read seed
\[
\mathrm{DR}(p,A)
\triangleq
\set*{t\in T^\prime\mid \delta(\delta(p,A),t)\ne\emptyset}.
\]
Define \(E_R\subseteq D\times D\) by
\[
(p,A)\leadsto_R(r,C)
\Longleftrightarrow
\delta(p,A)=r
\land C\in N^\prime
\land \mathrm{Null}(C)
\land (r,C)\in D.
\]
If \((r,C)\notin D\), then \(\Read\mathbin{!}(r,C)=\emptyset\), so no edge is
needed.  Compute
\[
\Read\mathbin{!}(p,A)
\triangleq
\mathrm{DIGRAPH}(D,E_R,\mathrm{DR})(p,A).
\]

\subsection{\texorpdfstring{Instance for \(\Follow\)}{Instance for Follow}}

The seed for follow propagation is the already computed read set:
\[
\mathrm{Seed}_F(p,A)\triangleq\Read\mathbin{!}(p,A).
\]
Define \(E_I\subseteq D\times D\) by
\[
(p,A)\leadsto_I(p^\prime,B)
\Longleftrightarrow
\Item{B}{\beta\cdot A\gamma}\in p
\land
\Path{\beta}{p^\prime}{p}
\land
\mathrm{Null}(\gamma)
\land
(p^\prime,B)\in D.
\]
The orientation means
\[
\Follow\mathbin{!}(p^\prime,B)\subseteq\Follow\mathbin{!}(p,A).
\]
Compute
\[
\Follow\mathbin{!}(p,A)
\triangleq
\mathrm{DIGRAPH}(D,E_I,\mathrm{Seed}_F)(p,A).
\]

\subsection{Lookback Relation and Lookahead Construction}

For every completed item \(\Item{A}{\omega\cdot\eps}\in q\), define
\[
\mathrm{LB}(q,A\to\omega)
\triangleq
\set*{(p,A)\in D\mid \Path{\omega}{p}{q}}.
\]
Then the implemented digraph guard is
\[
\widehat{\Look}(q,\Item{A}{\omega})
\triangleq
\bigcup_{(p,A)\in\mathrm{LB}(q,A\to\omega)}
\Follow\mathbin{!}(p,A),
\]
which is \eqref{eq:la-follow-formula} written as an implementation step.  It is
always a safe over-approximation of the semantic \(\Look\), and it is exact under
the productivity hypothesis of Corollary~\ref{cor:lookahead-follow}.

\paragraph{Implementation note.}
The query \(\Path{\omega}{p}{q}\) can be implemented by following
\(\delta\)-edges from \(p\) along \(\omega\).  The query
\(\Path{\beta}{p^\prime}{p}\) in \(E_I\) can be implemented by caching
reverse-path queries keyed by \((\beta,p)\), or by storing predecessor edges
during construction of the LR(0) automaton.

\subsection{Parser-Table and Conflict Specification}

After a guard \(K\) has been chosen---for instance \(K=\widehat{\Look}\) or, under productivity, \(K=\Look\)---construct actions as follows.
\begin{enumerate}
\item If \(\delta(q,t)\ne\emptyset\) for \(t\in T^\prime\), add the shift action
\[
q\xrightarrow{t}\delta(q,t).
\]
\item For each \(\Item{A}{\omega\cdot\eps}\in q\) and each
\(t\in K(q,\Item{A}{\omega})\), add the reduce action \(A\to\omega\) under
lookahead \(t\).
\item Report a shift/reduce conflict when both a shift action and a reduce
action are present for the same pair \((q,t)\).
\item Report a reduce/reduce conflict when two distinct reduce productions are
present for the same pair \((q,t)\).
\end{enumerate}
The accepting condition remains the one in Definition~\ref{def:lra}: the parser
accepts when it reaches
\[
(\Path{S\$}{q_0}{q_f},\eps).
\]
Equivalently, in a conventional ACTION/GOTO table, \(q_f\) may be treated as the
accepting state after the input marker has been consumed.

\subsection{Summary Specification}

The complete computation required by Theorem~\ref{thm:read-follow} is:
\begin{enumerate}
\item Build \(G^\prime\), \(Q\), and \(\delta\) as in
Definition~\ref{def:lra}.
\item Compute nullable nonterminals and nullable strings.
\item Build \(D\), \(\mathrm{DR}\), and \(E_R\); compute \(\Read\) by
\(\mathrm{DIGRAPH}(D,E_R,\mathrm{DR})\).
\item Build \(E_I\); compute \(\Follow\) by
\(\mathrm{DIGRAPH}(D,E_I,\Read)\).
\item Build \(\mathrm{LB}\) and compute each
\(\widehat{\Look}(q,\Item{A}{\omega})\) by unioning the appropriate
\(\Follow\)-sets.
\item Construct the guarded reduce actions and check conflicts.
\end{enumerate}
The postcondition is that the computed \(\Read\), \(\Follow\), and \(\widehat{\Look}\) are
the least sets described by Theorem~\ref{thm:read-follow}.  For arbitrary
grammars, \(\widehat{\Look}\) safely over-approximates the semantic lookahead and
therefore preserves the language when used as the guard in Section~3.  Under the
explicit productivity hypothesis, \(\widehat{\Look}=\Look\), so the computed sets are
exactly the semantic \(\LALRone\) lookahead sets determined by the LR(0) automaton.

\section{Fuel-Free Parser Execution}\label{sec:fuel-free-parser}

The table specification above is relational.  An executable parser can be
obtained without adding a fuel argument exactly when no reachable reduce-only
phase can run forever.  This section states that condition using the actual
parser stacks, not a finite control-only over-approximation.

\begin{definition}[Run stacks and reachable phase heads]\label{def:run-stacks-phase}
A run stack is a viable stack path starting at the initial LR state:
\[
\mathbf{Stack}
\triangleq
\set*{\Path{\alpha}{q_0}{q}
\mid
\alpha\in V^{\prime *},\ q\in Q,\ \Path{\alpha}{q_0}{q}}.
\]
The corresponding run configurations are
\[
\Config_{\mathrm{run}}
\triangleq
\set*{(s,u)
\mid
s\in\mathbf{Stack},\ u\in T^{\prime *}}.
\]
For an input word \(w\in T^*\), put
\[
c_0(w)
\triangleq
(\Path{\eps}{q_0}{q_0},w\$).
\]
A run configuration is reachable when
\[
\mathsf{Reach}(c)
\quad\Longleftrightarrow\quad
\exists w\in T^*.\ c_0(w)\vdash_{\Look}^{\,*}c.
\]
For \(s\in\mathbf{Stack}\) and \(t\in T^\prime\), say that \((s,t)\) is a
reachable reduce-phase head when
\[
\mathsf{Phase}(s,t)
\quad\Longleftrightarrow\quad
\exists u\in T^{\prime *}.\ \mathsf{Reach}(s,t u).
\]
\end{definition}

\begin{definition}[Exact reduce-phase relation]\label{def:exact-reduce-phase}
For each lookahead \(t\in T^\prime\), define a relation
\(\Rightarrow_t\subseteq\mathbf{Stack}\times\mathbf{Stack}\) by
\[
s\Rightarrow_t s^\prime
\]
iff there exist \(\alpha\in V^{\prime *}\), \(A\in N^\prime\),
\(\omega\in V^{\prime *}\), and states \(p,q,q^\prime\in Q\) such that
\[
s=\bigl(\Path{\alpha\omega}{q_0}{q}\bigr),
\qquad
s^\prime=\bigl(\Path{\alpha A}{q_0}{q^\prime}\bigr),
\]
\[
\Path{\alpha}{q_0}{p},
\qquad
\Path{\omega}{p}{q},
\qquad
\Item{A}{\omega}\in\Reduce_{\Look}(q,t),
\qquad
q^\prime=\delta(p,A).
\]
Thus \(s\Rightarrow_t s^\prime\) records one actual guarded Reduce step with
lookahead \(t\), including the stack prefix that is exposed by popping the
handle.
\end{definition}

\begin{lemma}[Exactness of reduce phases]\label{lem:exact-reduce-phase}
For \(s,s^\prime\in\mathbf{Stack}\), \(t\in T^\prime\), and
\(u\in T^{\prime *}\), the following are equivalent:
\[
(s,t u)\vdash_{\Look}(s^\prime,t u)
\quad\text{by a Reduce step}
\]
and
\[
s\Rightarrow_t s^\prime.
\]
Consequently \(\Rightarrow_t\) is neither forgetting the pop predecessor nor
combining incompatible stack contexts.
\end{lemma}

\begin{proof}
Expanding the guarded Reduce rule, a Reduce step from \((s,t u)\) to
\((s^\prime,t u)\) has the form
\[
s=\bigl(\Path{\alpha\omega}{q_0}{q}\bigr),
\qquad
s^\prime=\bigl(\Path{\alpha A}{q_0}{q^\prime}\bigr),
\]
with witnesses \(p\in Q\), \(A\in N^\prime\), and
\(\omega\in V^{\prime *}\) satisfying
\[
\Path{\alpha}{q_0}{p},
\qquad
\Path{\omega}{p}{q},
\qquad
\Item{A}{\omega}\in\Reduce_{\Look}(q,t),
\qquad
q^\prime=\delta(p,A).
\]
These are exactly the witnesses required by
Definition~\ref{def:exact-reduce-phase}.  The converse is the same argument read
backwards.
\end{proof}

\begin{definition}[Exact reduce-phase termination]\label{def:exact-phase-termination}
For a relation \(R\), write \(\mathrm{Acc}_R(x)\) for accessibility of \(x\): every
\(R\)-successor of \(x\) is again accessible.  The guarded table is
\emph{phase-terminating} when every reachable reduce-phase head is accessible for
the exact reduce relation:
\[
\forall s\in\mathbf{Stack},\ t\in T^\prime.\quad
\mathsf{Phase}(s,t)
\Longrightarrow
\mathrm{Acc}_{\Rightarrow_t}(s),
\]
where \(\mathrm{Acc}_{\Rightarrow_t}(s)\) uses the successor relation
\(s\Rightarrow_t s^\prime\).  This is the constructive well-foundedness
formulation.  The classically equivalent ``there is no infinite reachable
\(\Rightarrow_t\)-chain'' formulation is not used as the definition.
\end{definition}

\begin{lemma}[Finite branching of exact reductions]\label{lem:exact-reduce-finite-branching}
For each \(s\in\mathbf{Stack}\) and \(t\in T^\prime\), the set
\[
\set*{s^\prime\in\mathbf{Stack}\mid s\Rightarrow_t s^\prime}
\]
is finite.  If the guarded table is conflict-free, this set has at most one
element.
\end{lemma}

\begin{proof}
Write \(s=\Path{\beta}{q_0}{q}\).  A successor of \(s\) under
\(\Rightarrow_t\) must be produced by some completed item
\(\Item{A}{\omega}\in\Reduce_{\Look}(q,t)\).  This set of items is finite.  For a
fixed \(A\to\omega\), the equality \(\beta=\alpha\omega\) determines at most one
prefix \(\alpha\), and the deterministic LR path from \(q_0\) along \(\alpha\)
determines at most one exposed state \(p\).  Then the successor state is forced
to be \(\delta(p,A)\).  Hence only finitely many successors can occur.  If the
table is conflict-free, at most one Reduce action is available at the pair
\((q,t)\), so there is at most one successor.
\end{proof}

\begin{lemma}[Heights for finitely branching accessible relations]
\label{lem:finitely-branching-height}
Let \(R\subseteq X\times X\) be finitely branching, with a finite successor list
for each \(x\).  If \(\mathrm{Acc}_R(x)\), then one can define, by well-founded
recursion on this accessibility proof,
\[
H_R(x)
\triangleq
\begin{cases}
0,&\text{if }x\text{ has no }R\text{-successor},\\[2pt]
1+\max\set*{H_R(y)\mid x\mathrel R y},&\text{otherwise.}
\end{cases}
\]
Moreover, if \(x\mathrel R y\), then \(H_R(y)<H_R(x)\).  Conversely, any map
\(H:X\to\mathbb N\) that strictly decreases along \(R\) gives
\(\mathrm{Acc}_R(x)\) for every \(x\).
\end{lemma}

\begin{proof}
Accessibility gives the recursive calls \(H_R(y)\) for every successor
\(x\mathrel R y\); finite branching supplies the finite maximum.  The definition
therefore immediately yields \(H_R(x)\ge 1+H_R(y)\) for each successor, so
\(H_R(y)<H_R(x)\).  Conversely, if \(H\) strictly decreases along \(R\), prove
\(\mathrm{Acc}_R(x)\) by induction on \(H(x)\): every successor has strictly
smaller rank and is accessible by the induction hypothesis.
\end{proof}

\begin{definition}[Exact termination certificate]\label{def:termination-cert}
For a conflict-free guarded table, a termination certificate is a rank function
\[
\rho:\Config_{\mathrm{run}}\to\mathbb N
\]
such that every reachable Shift or Reduce step strictly decreases
\[
\mu(c)\triangleq(|\mathrm{input}(c)|,\rho(c))\in\mathbb N\times\mathbb N
\]
in lexicographic order.  Concretely, whenever \(\mathsf{Reach}(c)\) and
\(c\vdash_{\Look}c^\prime\) is a Shift or Reduce step, one has
\[
\mu(c^\prime)<_{\mathrm{lex}}\mu(c).
\]
Accept and Error configurations make no recursive call.
\end{definition}

\begin{definition}[Recursive-call relation]\label{def:recursive-call-relation}
Let \(\mathsf{Call}\subseteq\Config_{\mathrm{run}}\times\Config_{\mathrm{run}}\) be the relation
\[
c^\prime\mathrel{\mathsf{Call}}c
\]
iff \(\mathsf{Reach}(c)\), \(c\vdash_{\Look}c^\prime\), and that one parser step is either
Shift or Reduce.  Accept and Error configurations have no \(\mathsf{Call}\)
successors.
\end{definition}

\begin{lemma}[Certificate gives accessibility]\label{lem:termination-cert-acc}
If a termination certificate \(\rho\) exists, then every reachable run
configuration \(c\) satisfies
\[
\mathrm{Acc}_{\mathsf{Call}}(c).
\]
Equivalently, the recursive parser may make recursive calls only to
\(\mathsf{Call}\)-smaller configurations.
\end{lemma}

\begin{proof}
Put \(\mu(c)=(|\mathrm{input}(c)|,\rho(c))\).  By
Definition~\ref{def:termination-cert}, every \(\mathsf{Call}\)-edge
\(c^\prime\mathrel{\mathsf{Call}}c\) satisfies
\[
\mu(c^\prime)<_{\mathrm{lex}}\mu(c).
\]
The lexicographic order on \(\mathbb N\times\mathbb N\) is well-founded, and
well-foundedness is preserved by inverse image along \(\mu\).  Hence the measure
order
\[
c^\prime\prec_\rho c
\quad\Longleftrightarrow\quad
\mu(c^\prime)<_{\mathrm{lex}}\mu(c)
\]
is well-founded on run configurations.  Since \(\mathsf{Call}\) is a subrelation
of \(\prec_\rho\) on reachable configurations, every reachable configuration is
accessible for \(\mathsf{Call}\).
\end{proof}

\begin{definition}[Canonical exact rank]\label{def:canonical-exact-rank}
Assume the guarded table is phase-terminating.  For \(s\in\mathbf{Stack}\) and
\(t\in T^\prime\), define \(H_t(s)\) as follows.  If \(\mathsf{Phase}(s,t)\), use
the accessibility proof supplied by Definition~\ref{def:exact-phase-termination}
and Lemma~\ref{lem:finitely-branching-height} for \(\Rightarrow_t\).  If
\(\neg\mathsf{Phase}(s,t)\), put \(H_t(s)=0\).  For a run configuration
\(c=(s,u)\), put
\[
\rho_{\mathrm{ex}}(c)
\triangleq
\begin{cases}
H_t(s),&u=t u^\prime\text{ for some }t\in T^\prime,
                         u^\prime\in T^{\prime *},\\[2pt]
0,&u=\eps.
\end{cases}
\]
This rank is a proof object: because \(\mathsf{Reach}\) and \(\mathsf{Phase}\)
contain existential witnesses, \(\rho_{\mathrm{ex}}\) is not by itself a complete
decision procedure for arbitrary tables.
\end{definition}

\begin{theorem}[Sound and complete fuel-free criterion]\label{thm:exact-termination-criterion}
For a conflict-free guarded table, the following implications hold
constructively:
\begin{enumerate}
\item[(a)] phase-termination in the sense of
Definition~\ref{def:exact-phase-termination} implies the existence of a
termination certificate;
\item[(b)] a termination certificate implies that every public parser run
starting from some \(c_0(w)\), \(w\in T^*\), is finite.
\end{enumerate}
Conversely, in classical mathematics, finite public runs imply
phase-termination; hence the three conditions are equivalent classically.  When
phase-termination holds, the canonical exact rank \(\rho_{\mathrm{ex}}\) of
Definition~\ref{def:canonical-exact-rank} is a termination certificate.
\end{theorem}

\begin{proof}
\emph{Phase-termination implies a certificate.}
Assume phase-termination.  Let \(c=(s,u)\) be reachable.  If
\(u=t u^\prime\), then \(\mathsf{Phase}(s,t)\).  By
Definition~\ref{def:exact-phase-termination}, \(s\) is accessible for
\(\Rightarrow_t\); Lemma~\ref{lem:exact-reduce-finite-branching} and
Lemma~\ref{lem:finitely-branching-height} give a height \(H_t\) that strictly
decreases along every exact reduce edge.

We prove that \(\rho_{\mathrm{ex}}\) is a termination certificate.  If the
recursive step from \(c\) to \(c^\prime\) is a Shift and
\(\mathrm{input}(c)=t u^\prime\), then \(\mathrm{input}(c^\prime)=u^\prime\), so
\[
|\mathrm{input}(c^\prime)|<|\mathrm{input}(c)|.
\]
Therefore \(\mu(c^\prime)<_{\mathrm{lex}}\mu(c)\), independently of the second
component.  If the recursive step is a Reduce, write
\[
c=(s,t u^\prime),
\qquad
c^\prime=(s^\prime,t u^\prime).
\]
By Lemma~\ref{lem:exact-reduce-phase}, \(s\Rightarrow_t s^\prime\).  The height
decrease gives
\[
\rho_{\mathrm{ex}}(c^\prime)=H_t(s^\prime)<H_t(s)=\rho_{\mathrm{ex}}(c).
\]
The input length is unchanged by a Reduce step, so again
\(\mu(c^\prime)<_{\mathrm{lex}}\mu(c)\).  Accept and Error configurations make no
recursive call.  Thus \(\rho_{\mathrm{ex}}\) is a termination certificate.

\emph{A certificate implies finite public runs.}
Assume a termination certificate \(\rho\) exists.  Along every reachable
recursive parser step, the measure
\[
\mu(c)=(|\mathrm{input}(c)|,\rho(c))
\]
strictly decreases in the lexicographic order on \(\mathbb N\times\mathbb N\).
This order is well-founded, so no infinite sequence of reachable recursive calls
can start from any \(c_0(w)\).  Hence every public parser run is finite.

\emph{Classical converse.}
Assume classical dependent choice, or equivalently the standard finitely
branching form of K\H{o}nig's lemma.  If phase-termination failed, then some
reachable phase head \((s_0,t)\) would be non-accessible; finite branching would
produce an infinite chain
\[
s_0\Rightarrow_t s_1\Rightarrow_t s_2\Rightarrow_t\cdots .
\]
By \(\mathsf{Phase}(s_0,t)\), choose \(u\) and \(w\) with
\(c_0(w)\vdash_{\Look}^{*}(s_0,t u)\).  Lemma~\ref{lem:exact-reduce-phase} turns
each edge of the chain into a guarded Reduce step on the same unread input
\(t u\), so appending the infinite reduce phase to the finite prefix gives an
infinite public parser run.  Therefore, classically, finite public runs imply
phase-termination.
\end{proof}


\begin{corollary}[Fuel-free recursive recognizer]\label{cor:fuel-free-parser}
If the guarded table has a termination certificate, the public recognizer can be
exposed without a fuel parameter:
\[
\mathrm{run\_parser}:T^*\to\mathrm{bool}.
\]
It returns \(\mathrm{true}\) exactly for accepted inputs.  A parser returning
\(\mathrm{option}\ \mathrm{ParseTree}\) requires a separate tree-carrying
configuration type and a proof that the produced tree denotes the corresponding
derivation; that structure is not part of the recognizer defined in this note.
\end{corollary}

\begin{proof}
Use the certificate rank to order recursive calls by
\[
c^\prime\prec c
\quad\Longleftrightarrow\quad
\mu(c^\prime)<_{\mathrm{lex}}\mu(c).
\]
The lexicographic order on \(\mathbb N\times\mathbb N\) is well-founded, and the
certificate supplies the required decrease for every reachable Shift or Reduce
call.  Thus the internal runner is defined on reachable configurations by
well-founded recursion on \(\prec\), equivalently by carrying a reachability
proof internally.  The initial configuration \(c_0(w)\) is reachable by the empty
run, so the public wrapper supplies the accessibility proof and exposes no fuel
argument.  Its result is Boolean because the configurations above carry only
control and input, not parse-tree fragments.
\end{proof}

The criterion above is exact as a mathematical specification: it rejects a table
only when there is an actual infinite reduce-only phase reachable from some
initial input.  It is not, by itself, a complete executable decision procedure,
because \(\mathsf{Reach}\), \(\mathsf{Phase}\), and the canonical rank
\(\rho_{\mathrm{ex}}\) contain existential proof data.  A finite control-only graph
may still be useful as a sufficient quick check, but it is not complete because
such a graph can contain cycles that no single parser stack can realise.  For an
executable Coq development one should either use such sufficient checks or supply
a table-specific termination certificate.
