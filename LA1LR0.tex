\begin{abstract}
\noindent
This note fixes the LR(0) characteristic automaton, defines the one-symbol
lookahead set attached to each completed LR(0) item, and derives the usual
Read/Follow digraph computation.  The proofs are written so that the essential
claims appear as inclusions, fixed-point equations, and derivation chains.
\end{abstract}

\section{The LR(0) Automaton}

\begin{definition}[$\LRA(G)$]\label{def:lra}
Let \(G \triangleq (N,T,P,S)\) be a context-free grammar.  Define:
\begin{enumerate}[label=\textup{(\roman*)}]
\item
\[
G^\prime=(N^\prime,T^\prime,P^\prime,S^\prime)
\triangleq
\bigl(N\uplus\set*{S^\prime},\,
      T\uplus\set*{\$},\,
      P\cup\set*{\Item{S^\prime}{S\$}},\,
      S^\prime\bigr).
\]

\item
\[
V\triangleq N\uplus T,\qquad
V^\prime\triangleq N^\prime\uplus T^\prime .
\]

\item
\[
\beta\RMP\gamma
\quad\xLeftrightarrow{\mathrm{def}}\quad
(\beta,\gamma)\in
\set*{(\alpha A z,\alpha\omega z)
\mid
\Item{A}{\omega}\in P^\prime,\ \alpha\in V^{\prime *},\ z\in T^{\prime *}}.
\]

\item
\[
I_{\LRzero}
\triangleq
\set*{\Item{A}{\alpha\cdot\beta}
\mid
\Item{A}{\alpha\beta}\in P^\prime}.
\]

\item For \(q\subseteq I_{\LRzero}\),
\[
\mathrm{Cl}(q)
=_\mu
q\cup
\set*{\Item{A}{\eps\cdot\omega}
\mid
\Item{A}{\omega}\in P^\prime,\
\Item{B}{\beta\cdot A\gamma}\in \mathrm{Cl}(q)}.
\]

\item
\[
q_0\triangleq
\mathrm{Cl}\bigl(\set*{\Item{S^\prime}{\eps\cdot S\$}}\bigr).
\]

\item For \(q\subseteq I_{\LRzero}\) and \(X\in V^\prime\),
\[
\mathrm{GOTO}(q,X)
\triangleq
\mathrm{Cl}\bigl(
\set*{\Item{A}{\alpha X\cdot\beta}
\mid
\Item{A}{\alpha\cdot X\beta}\in q}
\bigr).
\]

\item
\[
Q\triangleq \mathrm{PT}\setminus\set*{\emptyset},
\qquad
\mathrm{PT}
=_\mu
\set*{q_0}
\cup
\set*{\mathrm{GOTO}(q,X)\mid q\in\mathrm{PT},\ X\in V^\prime}.
\]

\item The path relation is the least relation satisfying
\[
\eps:p\to q \Longleftrightarrow p\in Q\land p=q,
\qquad
X\alpha:p\to q
\Longleftrightarrow
p\in Q\land \alpha:\mathrm{GOTO}(p,X)\to q .
\]

\item
\[
\Config
\triangleq
\set*{(\Path{\alpha}{p}{q},z)
\mid
\alpha\in V^{\prime *},\ p,q\in Q,\ z\in T^{\prime *},\
\Path{\alpha}{p}{q}}.
\]

\item
\[
\delta(q,X)\triangleq \mathrm{GOTO}(q,X)
\qquad(q\in Q,\ X\in V^\prime).
\]

\item
\[
\Reduce(q,t)
\triangleq
\set*{\Item{A}{\omega}
\mid
\Item{A}{\omega\cdot\eps}\in q}
\qquad(q\in Q,\ t\in T^\prime).
\]

\item Let \({\vdash}\subseteq\Config\times\Config\) be generated by:
\[
\infer[\mathrm{Shift}]
  {(\Path{\alpha}{q}{q^\prime},tz)
    \vdash
   (\Path{\alpha t}{q}{q^{\prime\prime}},z)}
  {q^{\prime\prime}=\delta(q^\prime,t)}
\]
\[
\infer[\mathrm{Reduce}(A\to\omega)]
  {(\Path{\alpha\omega}{q}{q^\prime},tz)
    \vdash
   (\Path{\alpha A}{q}{q^{\prime\prime}},tz)}
  {\Path{\alpha}{q}{p}
   &
   \Path{\omega}{p}{q^\prime}
   &
   \Item{A}{\omega}\in\Reduce(q^\prime,t)
   &
   q^{\prime\prime}=\delta(p,A)}
\]

\item
\[
q_f\triangleq\delta(\delta(q_0,S),\$).
\]

\item The accepted language is
\[
L(\LRA(G))
\triangleq
\set*{z\in T^*
\mid
(\Path{\eps}{q_0}{q_0},z\$)
\vdash^*
(\Path{S\$}{q_0}{q_f},\eps)}.
\]
\end{enumerate}
\end{definition}

\begin{fact}[Elimination of non-productive nonterminals]\label{fact:productive}
Call \(A\in N^\prime\) \emph{productive} if \(A\Rightarrow_{P^\prime}^*w\) for some
\(w\in T^{\prime *}\); equivalently \(A\in\mathrm{Gen}\), the least set with
\[
A\in\mathrm{Gen}
\iff
\exists\,\Item{A}{X_1\cdots X_k}\in P^\prime.\
\forall i\in\set*{1,\ldots,k}.\ X_i\in T^\prime\cup\mathrm{Gen}
\qquad(k=0\text{ allowed}).
\]
Otherwise \(A\) is \emph{non-productive}.  A production that mentions a
non-productive nonterminal can never occur in a derivation of a terminal string,
so deleting every non-productive nonterminal together with all productions that
mention one preserves the generated language.  Hence, whenever
\(L(G)\ne\emptyset\), we may assume without loss of generality that every
nonterminal of \(G^\prime\) is productive; when \(L(G)=\emptyset\) the identity
\(L(\LRA(G))=L(G)=\emptyset\) already follows from soundness
(Corollary~\ref{cor:lra-sound}).
\end{fact}

For \(c=(\Path{\alpha}{p}{q},u)\in\Config\), put
\[
\mathrm{Y}(c)\triangleq\alpha u\in V^{\prime *}.
\]

The path relation factors through the state reached after any prefix: induction
on \(|\alpha|\) using Definition~\ref{def:lra}(ix) gives, for all
\(\alpha,\omega\in V^{\prime *}\) and \(r\in Q\),
\begin{equation}\label{eq:path-concat}
\Path{\alpha\omega}{r}{q}
\iff
\exists!\,p\in Q.\ \Path{\alpha}{r}{p}\ \land\ \Path{\omega}{p}{q},
\end{equation}
the intermediate state being \(p=\delta(\cdots\delta(\delta(r,\alpha_1),\alpha_2)\cdots,\alpha_{|\alpha|})\)
since each \(\delta(\cdot,X)=\mathrm{GOTO}(\cdot,X)\) is single-valued.

\begin{lemma}[LR(0) path invariants]\label{lem:lra-path-invariants}
For \(\eta\in V^{\prime *}\), the characteristic automaton satisfies
\begin{align}
(\exists q\in Q)\ \Path{\eta}{q_0}{q}
&\Longleftrightarrow
\eta\text{ is a viable prefix of }G^\prime, \label{eq:vp}\\
\left.
\begin{gathered}
\eta=\eta^\prime\omega,\quad
\Path{\eta^\prime}{q_0}{p},\quad
\Path{\omega}{p}{q},\\
S^\prime\RMP^*\eta^\prime A y
\RMP
\eta^\prime\omega y,\quad y=t z\in T^{\prime *}
\end{gathered}
\right\}
&\Longrightarrow
\Item{A}{\omega\cdot\eps}\in q
\ \land\
\Item{A}{\omega}\in\Reduce(q,t).
\label{eq:handle}
\end{align}
\end{lemma}

\begin{proof}
By Fact~\ref{fact:productive} we may assume every nonterminal of \(G^\prime\) is
productive: \(C\Rightarrow_{P^\prime}^*x\) for some \(x\in T^{\prime *}\).  Write
\(\sqsubseteq\) for the prefix order on
\(V^{\prime *}\).  Call an item \emph{valid} for \(\eta\in V^{\prime *}\) when
\begin{equation}\label{eq:valid-def}
\Item{A}{\alpha\cdot\beta}\in\mathsf{Valid}(\eta)
\ \overset{\mathrm{def}}{\iff}\
\exists\,\delta\in V^{\prime *},\,w\in T^{\prime *}.\
\eta=\delta\alpha\ \land\ S^\prime\RMP^*\delta Aw
\end{equation}
(then \(\Item{A}{\alpha\beta}\in P^\prime\) and \(w\in T^{\prime *}\) make
\(\delta Aw\RMP\delta\alpha\beta w\) a rightmost step).  Validity captures
viability:
\begin{equation}\label{eq:valid-vp}
\mathsf{Valid}(\eta)\ne\emptyset
\iff
\eta\text{ is a viable prefix of }G^\prime,
\end{equation}
since \(\Item{A}{\alpha\cdot\beta}\in\mathsf{Valid}(\eta)\) yields
\(S^\prime\RMP^*\delta Aw\RMP\delta\alpha\beta w\) with
\(\eta=\delta\alpha\sqsubseteq\delta\alpha\beta\), while a viable prefix
\(\eta\sqsubseteq\delta\alpha\beta\) (from
\(S^\prime\RMP^*\delta Aw\RMP\delta\alpha\beta w\)) factors as
\(\eta=\delta\alpha\beta_1\), \(\beta=\beta_1\beta_2\), giving
\(\Item{A}{\alpha\beta_1\cdot\beta_2}\in\mathsf{Valid}(\eta)\).

\proofpart{Closure computes the valid items at a fixed \(\eta\)}
If \(q\subseteq\mathsf{Valid}(\eta)\) contains every valid item with nonempty
left part (and the seed \(\Item{S^\prime}{\eps\cdot S\$}\) when \(\eta=\eps\)),
then \(\mathrm{Cl}(q)=\mathsf{Valid}(\eta)\).  For \(\subseteq\), the closure rule
adds \(\Item{A}{\eps\cdot\omega}\) only from some
\(\Item{B}{\beta_1\cdot A\beta_2}\in\mathrm{Cl}(q)\subseteq\mathsf{Valid}(\eta)\),
and productivity \(\beta_2\RMP^*x\in T^{\prime *}\) gives
\[
S^\prime\RMP^*\delta Bw\RMP\delta\beta_1A\beta_2w\RMP^*\delta\beta_1Axw,
\qquad
\eta=\delta\beta_1,\ \ xw\in T^{\prime *}
\ \Longrightarrow\
\Item{A}{\eps\cdot\omega}\in\mathsf{Valid}(\eta).
\]
For \(\supseteq\), take \(\Item{A}{\eps\cdot\omega}\in\mathsf{Valid}(\eta)\) with a
witness \(S^\prime\RMP^*\eta Aw\) and induct on its length.  Length \(0\) forces
\(\eta Aw=S^\prime\), i.e.\ \(\eta=\eps\), \(A=S^\prime\), the seed.  Otherwise the
step first introducing this occurrence of \(A\) reads
\[
S^\prime\RMP^*\mu C\rho\RMP\mu\sigma_1A\sigma_2\rho,
\qquad
\Item{C}{\sigma_1A\sigma_2}\in P^\prime,\quad
\eta=\mu\sigma_1,\quad \rho\in T^{\prime *},
\]
so \(\Item{C}{\sigma_1\cdot A\sigma_2}\in\mathsf{Valid}(\eta)\) by the shorter
witness \(S^\prime\RMP^*\mu C\rho\).  This item lies in \(q\) (if
\(\sigma_1\ne\eps\)) or in \(\mathrm{Cl}(q)\) (induction hypothesis, if
\(\sigma_1=\eps\)); its dot precedes \(A\), so the closure rule appends
\(\Item{A}{\eps\cdot\omega}\in\mathrm{Cl}(q)\).

\proofpart{\(\mathrm{GOTO}\) shifts the prefix}
By \eqref{eq:valid-def}, for every \(X\in V^\prime\),
\[
\Item{A}{\alpha\cdot X\beta}\in\mathsf{Valid}(\eta)
\iff
\Item{A}{\alpha X\cdot\beta}\in\mathsf{Valid}(\eta X)
\qquad(\text{same }\delta,w;\ \eta=\delta\alpha\iff\eta X=\delta\alpha X),
\]
and every nonempty-left-part valid item for \(\eta X\) ends in \(X\), hence has
this form.  So the kernel of \(\mathrm{GOTO}(\mathsf{Valid}(\eta),X)\) is exactly
those items, and closing it gives
\begin{equation}\label{eq:goto-valid}
q=\mathsf{Valid}(\eta)\ \Longrightarrow\ \mathrm{GOTO}(q,X)=\mathsf{Valid}(\eta X).
\end{equation}

\proofpart{Induction on \(|\eta|\)}
No valid item for \(\eps\) has a nonempty left part, so
\(q_0=\mathrm{Cl}\bigl(\set*{\Item{S^\prime}{\eps\cdot S\$}}\bigr)
=\mathsf{Valid}(\eps)\).  Inductively, \(\Path{\eta X}{q_0}{q^\prime}\) means
\(\Path{\eta}{q_0}{q}\) with \(q=\mathsf{Valid}(\eta)\) and
\(q^\prime=\delta(q,X)=\mathrm{GOTO}(q,X)\overset{\eqref{eq:goto-valid}}{=}\mathsf{Valid}(\eta X)\).
Hence
\begin{equation}\label{eq:state-eq-valid}
\Path{\eta}{q_0}{q}\ \Longrightarrow\ q=\mathsf{Valid}(\eta).
\end{equation}
Because every state on a path lies in \(Q=\mathrm{PT}\setminus\set*{\emptyset}\),
and prefixes of a viable prefix are viable (so a run never blocks),
\[
(\exists q\in Q)\,\Path{\eta}{q_0}{q}
\overset{\eqref{eq:state-eq-valid}}{\iff}
\mathsf{Valid}(\eta)\ne\emptyset
\overset{\eqref{eq:valid-vp}}{\iff}
\eta\text{ is a viable prefix of }G^\prime,
\]
which is \eqref{eq:vp}.

\proofpart{Handle invariant}
With \(\eta=\eta^\prime\omega\), \(\Path{\eta^\prime}{q_0}{p}\),
\(\Path{\omega}{p}{q}\), and \(S^\prime\RMP^*\eta^\prime Ay\RMP\eta^\prime\omega y\),
\(y=tz\in T^{\prime *}\): \eqref{eq:path-concat} gives
\(\Path{\eta^\prime\omega}{q_0}{q}\), so
\(q=\mathsf{Valid}(\eta^\prime\omega)\) by \eqref{eq:state-eq-valid}.  Taking
\(\delta=\eta^\prime\), \(w=y\) in \eqref{eq:valid-def},
\[
S^\prime\RMP^*\eta^\prime Ay,\ \ y\in T^{\prime *}
\ \Longrightarrow\
\Item{A}{\omega\cdot\eps}\in\mathsf{Valid}(\eta^\prime\omega)=q
\ \overset{\text{Def.\,\ref{def:lra}(xii)}}{\Longrightarrow}\
\Item{A}{\omega}\in\Reduce(q,t),
\]
which is \eqref{eq:handle}.
\end{proof}

\begin{lemma}[One-step yield invariant]\label{lem:lra-one-step-yield}
Each parser step reverses zero or one rightmost grammar step:
\begin{equation}\label{eq:one-step-yield}
c\vdash c'
\Longrightarrow
\begin{cases}
\mathrm{Y}(c')=\mathrm{Y}(c),
  &\text{if the step is Shift},\\[2pt]
\mathrm{Y}(c')\RMP\mathrm{Y}(c),
  &\text{if the step is Reduce}.
\end{cases}
\end{equation}
\end{lemma}

\begin{proof}
Indeed,
\[
\begin{array}{rclcl}
\mathrm{Shift}:&
\mathrm{Y}(\Path{\alpha}{q}{q^\prime},tz)
&=&\alpha tz
&=\mathrm{Y}(\Path{\alpha t}{q}{q^{\prime\prime}},z),\\[2pt]
\mathrm{Reduce}:&
\mathrm{Y}(\Path{\alpha A}{q}{q^{\prime\prime}},tz)
&=&\alpha A t z
&\RMP \alpha\omega t z
=\mathrm{Y}(\Path{\alpha\omega}{q}{q^\prime},tz).
\end{array}
\]
\end{proof}

\begin{corollary}[Multi-step yield invariant]\label{cor:lra-star-yield}
For \(c,c'\in\Config\),
\begin{equation}\label{eq:star-yield}
c\vdash^* c'
\Longrightarrow
\mathrm{Y}(c')\RMP^*\mathrm{Y}(c).
\end{equation}
\end{corollary}

\begin{proof}
Induct on the length of \(c\vdash^*c'\) and use
Lemma~\ref{lem:lra-one-step-yield} at each parser step.
\end{proof}

\begin{corollary}[LR(0) soundness]\label{cor:lra-sound}
\[
L(\LRA(G))\subseteq L(G).
\]
\end{corollary}

\begin{proof}
\[
\begin{aligned}
z\in L(\LRA(G))
&\Longrightarrow
(\Path{\eps}{q_0}{q_0},z\$)
\vdash^*
(\Path{S\$}{q_0}{q_f},\eps)\\
&\overset{\eqref{eq:star-yield}}{\Longrightarrow}
S\$\RMP^*z\$\\
&\Longrightarrow
S\DerP^*z\\
&\Longrightarrow
z\in L(G).
\end{aligned}
\]
The third implication deletes the endmarker.  Write the rightmost derivation
witnessing \(S\$\RMP^*z\$\) as
\[
S\$=\gamma_0\RMP\gamma_1\RMP\cdots\RMP\gamma_M=z\$ .
\]
Since \(S^\prime\) occurs on no right-hand side of \(P^\prime\), the only
production mentioning \(S^\prime\), namely \(\Item{S^\prime}{S\$}\), is never
applicable to a form derived from \(S\$\); hence every step uses a production of
\(P\subseteq P^\prime\).  Moreover \(\$\notin V\) occurs on no right-hand side of
\(P\), and no production of \(P\) rewrites a terminal, so the rightmost symbol
\(\$\) of \(\gamma_0\) is never touched:
\[
\gamma_k=\delta_k\$,\qquad \delta_k\in V^{*}\quad(0\le k\le M),
\qquad \delta_0=S,\quad \delta_M=z,
\qquad \delta_k\DerP\delta_{k+1}.
\]
Stripping the common trailing \(\$\) therefore yields
\(S=\delta_0\DerP\delta_1\DerP\cdots\DerP\delta_M=z\), i.e.\ \(S\DerP^*z\).
\end{proof}

\begin{lemma}[LR(0) completeness]\label{lem:lra-complete}
\[
L(G)\subseteq L(\LRA(G)).
\]
\end{lemma}

\begin{proof}
Let \(z\in L(G)\), so \(S\DerP^*z\).  Choose a rightmost derivation in \(G^\prime\):
\begin{equation}\label{eq:rightmost-chain}
S\$=\gamma_0\RMP\gamma_1\RMP\cdots\RMP\gamma_N=z\$.
\end{equation}
Write its \(k\)-th step as
\begin{equation}\label{eq:k-step}
\gamma_k=\alpha_kA_k y_k
\RMP
\alpha_k\omega_k y_k=\gamma_{k+1},
\qquad
\Item{A_k}{\omega_k}\in P^\prime,\quad y_k=t_k z_k\in T^{\prime *}.
\end{equation}
By \eqref{eq:handle}, for each \(k\) there exist \(p_k,q_k\in Q\) such that
\begin{equation}\label{eq:k-handle}
\Path{\alpha_k}{q_0}{p_k},
\qquad
\Path{\omega_k}{p_k}{q_k},
\qquad
\Item{A_k}{\omega_k\cdot\eps}\in q_k,
\qquad
\Item{A_k}{\omega_k}\in\Reduce(q_k,t_k).
\end{equation}
Reading \eqref{eq:rightmost-chain} from right to left turns each step into one
inverse reduction.  By \eqref{eq:k-handle} and \eqref{eq:path-concat},
\(\Path{\alpha_k\omega_k}{q_0}{q_k}\), so the \(\mathrm{Reduce}(A_k\to\omega_k)\)
rule applies:
\begin{equation}\label{eq:inverse-reduce}
(\Path{\alpha_k\omega_k}{q_0}{q_k},y_k)
\vdash
(\Path{\alpha_kA_k}{q_0}{\delta(p_k,A_k)},y_k)
\qquad(k=N-1,\ldots,0).
\end{equation}
These reductions are interleaved with Shift blocks.  Since \(y_k\in T^{\prime *}\),
the symbol \(A_k\) exposed at step \(k\) is the rightmost nonterminal of
\(\gamma_k=\alpha_kA_ky_k\); using the previous step
\(\gamma_k=\alpha_{k-1}\omega_{k-1}y_{k-1}\) with \(y_{k-1}\in T^{\prime *}\) and
cancelling the common terminal suffix,
\begin{equation}\label{eq:shift-block}
\alpha_kA_k\,u_k=\alpha_{k-1}\omega_{k-1},
\qquad
y_k=u_k\,y_{k-1},
\qquad
u_k\in T^{\prime *}
\qquad(1\le k\le N-1).
\end{equation}
By \eqref{eq:path-concat} the path \(\Path{\alpha_{k-1}\omega_{k-1}}{q_0}{q_{k-1}}\)
factors through \(\delta(p_k,A_k)\) after the prefix \(\alpha_kA_k\), so the
\(|u_k|\) symbols of \(u_k\) are shifted one by one:
\[
(\Path{\alpha_kA_k}{q_0}{\delta(p_k,A_k)},\,u_ky_{k-1})
\ \vdash^{|u_k|}\
(\Path{\alpha_{k-1}\omega_{k-1}}{q_0}{q_{k-1}},\,y_{k-1}).
\]
At the bottom (\(k=N-1\)), the run starts by shifting all of
\(\alpha_{N-1}\omega_{N-1}\) out of \(z\$=\gamma_N\):
\[
(\Path{\eps}{q_0}{q_0},z\$)
\ \vdash^{|\alpha_{N-1}\omega_{N-1}|}\
(\Path{\alpha_{N-1}\omega_{N-1}}{q_0}{q_{N-1}},y_{N-1}).
\]
At the top (\(k=0\)) we have \(\gamma_0=S\$\), whence \(\alpha_0=\eps\),
\(A_0=S\), \(y_0=\$\); \eqref{eq:inverse-reduce} leaves the stack \(S\) reading
\(\$\), and one final Shift of \(\$\) gives
\[
(\Path{S}{q_0}{\delta(q_0,S)},\$)
\ \vdash\
(\Path{S\$}{q_0}{q_f},\eps),
\qquad
q_f=\delta(\delta(q_0,S),\$).
\]
Concatenating the Shift blocks with \eqref{eq:inverse-reduce} for
\(k=N-1,\ldots,0\) gives
\[
(\Path{\eps}{q_0}{q_0},z\$)
\vdash^*
(\Path{S\$}{q_0}{q_f},\eps),
\]
hence \(z\in L(\LRA(G))\).
\end{proof}

\begin{corollary}[Correctness of \(\LRA(G)\)]\label{coro:lra-correct}
\[
L(G)=L(\LRA(G)),
\qquad
L(G)\triangleq\set*{z\in T^*\mid S\DerP^*z}.
\]
\end{corollary}

\begin{proof}
Corollary~\ref{cor:lra-sound} and Lemma~\ref{lem:lra-complete} give
\[
L(\LRA(G))\subseteq L(G),
\qquad
L(G)\subseteq L(\LRA(G)).
\]
\end{proof}

\section{Read and Follow Sets}

Define
\[
\Look:
\set*{(q,\Item{A}{\omega})
\mid q\in Q,\ \Item{A}{\omega\cdot\eps}\in q}
\longrightarrow \Pow(T^\prime)
\]
by
\begin{equation}\label{eq:la-def}
\Look(q,\Item{A}{\omega})
\triangleq
\set*{t\in T^\prime
\mid
S^\prime\RMP^*\alpha A t z,\
\Path{\alpha\omega}{q_0}{q},\
\alpha\in V^{\prime *},\
z\in T^{\prime *}}.
\end{equation}

The remainder of this section computes \(\Look(q,\Item{A}{\omega})\) from the LR(0) automaton, as the least fixed point of a Read/Follow digraph.

Let
\[
D\triangleq
\set*{(p,A)\mid p\in Q,\ A\in N^\prime,\ \delta(p,A)\ne\emptyset}.
\]
For a relation \(R\), write
\[
R\mathbin{!}x\triangleq\set*{y\mid (x,y)\in R}.
\]
Define \(\Read,\Follow\subseteq D\times T^\prime\) as the least relations
generated by the rules
\[
\begin{array}{cc}
\infer[\mathrm{DR}]
  {t\in\Read\mathbin{!}(p,A)}
  {\delta(\delta(p,A),t)\ne\emptyset}
&
\infer[\mathrm{reads}]
  {\Read\mathbin{!}(r,C)\subseteq\Read\mathbin{!}(p,A)}
  {\delta(p,A)=r & C\Rightarrow_{P^\prime}^*\eps & (r,C)\in D}
\\[4ex]
\infer[\mathrm{Read}]
  {t\in\Follow\mathbin{!}(p,A)}
  {t\in\Read\mathbin{!}(p,A)}
&
\infer[\mathrm{includes}]
  {\Follow\mathbin{!}(p^\prime,B)\subseteq\Follow\mathbin{!}(p,A)}
  {\Item{B}{\beta\cdot A\gamma}\in p
   & \Path{\beta}{p^\prime}{p}
   & \gamma\Rightarrow_{P^\prime}^*\eps
   & (p^\prime,B)\in D}
\end{array}
\]

Define the semantic read and follow sets on \(D\):
\begin{align}
\mathsf{Read}(p,A)
&\triangleq
\set*{t\in T^\prime
\mid
\exists r,s,\gamma.\ 
r=\delta(p,A)
\land
\gamma\Rightarrow_{P^\prime}^*\eps
\land
\gamma t:r\to s}, \label{eq:sem-read}\\
\mathsf{Follow}(p,A)
&\triangleq
\set*{t\in T^\prime
\mid
\exists\alpha,z.\ 
S^\prime\RMP^*\alpha A t z
\land
\Path{\alpha}{q_0}{p}
\land
z\in T^{\prime *}}. \label{eq:sem-follow}
\end{align}

For \((p,A)\in D\), set
\[
\mathrm{DR}(p,A)
\triangleq
\set*{t\in T^\prime\mid \delta(\delta(p,A),t)\ne\emptyset}.
\]
Let
\[
\Phi_R(F)(p,A)
=
\mathrm{DR}(p,A)
\cup
\bigcup_{\substack{\delta(p,A)=r\\ C\Rightarrow_{P^\prime}^*\eps\\(r,C)\in D}}
F(r,C).
\]

\begin{lemma}[Semantic Read fixed point]\label{lem:read-fixed-point}
For \((p,A)\in D\),
\begin{equation}\label{eq:read-eq}
\Read\mathbin{!}(p,A)
=
\mu\Phi_R(p,A)
=
\mathsf{Read}(p,A).
\end{equation}
\end{lemma}

\begin{proof}
The two \(\Read\)-rules say exactly that \(\Read\mathbin{!}\) is the least
\(\Phi_R\)-closed family, i.e.\ \(\Read\mathbin{!}(p,A)=\mu\Phi_R(p,A)\); since
\(\Phi_R\) is monotone on the finite lattice \(D\to\Pow(T^\prime)\),
\[
\mu\Phi_R=\bigcup_{m\ge 0}\Phi_R^{m}(\emptyset).
\]
Write \((p,A)\rightsquigarrow_R(r,C)\) for the reads edge
\(r=\delta(p,A)\land C\Rightarrow_{P^\prime}^*\eps\land(r,C)\in D\).  Because
\(\Phi_R(\emptyset)(p,A)=\mathrm{DR}(p,A)\), an induction on \(m\) unfolds the
operator into reads-paths:
\begin{equation}\label{eq:read-unfold}
\Phi_R^{m+1}(\emptyset)(p,A)
=\bigcup_{i=0}^{m}\
\bigcup_{(p,A)\rightsquigarrow_R^{\,i}(r,C)}
\mathrm{DR}(r,C).
\end{equation}
By \eqref{eq:sem-read}, \(t\in\mathsf{Read}(p,A)\) iff for some nullable
\(\gamma=C_1\cdots C_n\in N^{\prime *}\),
\[
r_0=\delta(p,A),\qquad
r_i=\delta(r_{i-1},C_i)\ (1\le i\le n),\qquad
\delta(r_n,t)\ne\emptyset ,
\]
which is precisely an \(n\)-edge reads-path ending at a \(\mathrm{DR}\)-seed
holding \(t\):
\[
(p,A)\rightsquigarrow_R(r_0,C_1)\rightsquigarrow_R\cdots\rightsquigarrow_R
(r_{n-1},C_n),
\qquad
t\in\mathrm{DR}(r_{n-1},C_n)
\quad\bigl(\delta(\delta(r_{n-1},C_n),t)=\delta(r_n,t)\ne\emptyset\bigr).
\]
By \eqref{eq:read-unfold} this is equivalent to
\(t\in\Phi_R^{\,n+1}(\emptyset)(p,A)\subseteq\mu\Phi_R(p,A)\), and conversely each
member of the union \eqref{eq:read-unfold} reads back as such a nullable
\(\delta\)-chain.  Hence \eqref{eq:read-eq}.
\end{proof}

Let
\[
\Phi_F(F)(p,A)
=
\mathsf{Read}(p,A)
\cup
\bigcup_{\substack{
  \Item{B}{\beta\cdot A\gamma}\in p\\
  \Path{\beta}{p^\prime}{p}\\
  \gamma\Rightarrow_{P^\prime}^*\eps\\
  (p^\prime,B)\in D}}
F(p^\prime,B).
\]

\begin{lemma}[Semantic Follow fixed point]\label{lem:follow-fixed-point}
For \((p,A)\in D\),
\begin{equation}\label{eq:follow-eq}
\Follow\mathbin{!}(p,A)
=
\mu\Phi_F(p,A)
=
\mathsf{Follow}(p,A).
\end{equation}
\end{lemma}

\begin{proof}
By Lemma~\ref{lem:read-fixed-point}, the two \(\Follow\)-rules define the least
fixed point of \(\Phi_F\).
First \(\mathsf{Follow}\) is closed under \(\Phi_F\).
For the \(\mathsf{Read}\)-seed, fix \(t\in\mathsf{Read}(p,A)\).  As \(p\in Q\) is
reachable from \(q_0\) by construction of \(\mathrm{PT}\), pick \(\alpha\) with
\(\Path{\alpha}{q_0}{p}\); by \eqref{eq:sem-read} there is a nullable \(\gamma\)
with \(\Path{\gamma t}{\delta(p,A)}{s}\), so \eqref{eq:path-concat} yields
\(\Path{\alpha A\gamma t}{q_0}{s}\), and \eqref{eq:vp} makes \(\alpha A\gamma t\)
a viable prefix, i.e.\ \(S^\prime\RMP^*\alpha A\gamma t z\) for some
\(z\in T^{\prime *}\).  Hence
\[
\begin{aligned}
t\in\mathsf{Read}(p,A)
&\Longrightarrow
\exists\alpha,\gamma,z.\
\Path{\alpha}{q_0}{p}
\land
\gamma\Rightarrow_{P^\prime}^*\eps
\land
S^\prime\RMP^*\alpha A\gamma t z\\
&\Longrightarrow
S^\prime\RMP^*\alpha A\gamma t z
\RMP^*
\alpha A t z
\qquad(\gamma\Rightarrow_{P^\prime}^*\eps\text{ rewritten rightmost})\\
&\Longrightarrow
t\in\mathsf{Follow}(p,A).
\end{aligned}
\]
For an includes edge,
\[
\left.
\begin{gathered}
t\in\mathsf{Follow}(p^\prime,B),\quad
\Item{B}{\beta\cdot A\gamma}\in p,\quad
\Path{\beta}{p^\prime}{p},\quad
\gamma\Rightarrow_{P^\prime}^*\eps
\end{gathered}
\right\}
\Longrightarrow
\begin{aligned}[t]
S^\prime&\RMP^*\alpha^\prime B t z\\
&\RMP \alpha^\prime\beta A\gamma t z\\
&\RMP^*\alpha^\prime\beta A t z,
\end{aligned}
\]
with \(\Path{\alpha^\prime\beta}{q_0}{p}\); hence
\[
\mathsf{Follow}(p^\prime,B)\subseteq\mathsf{Follow}(p,A).
\]

Conversely, let \(t\in\mathsf{Follow}(p,A)\), witnessed by
\[
S^\prime\RMP^*\alpha A t z,
\qquad
\Path{\alpha}{q_0}{p}.
\]
The production \(B\to\beta A\gamma\) that created this occurrence of \(A\) gives a
factorization
\[
S^\prime\RMP^*
\alpha^\prime B t^\prime z^\prime
\RMP
\alpha^\prime\beta A\gamma t^\prime z^\prime
\RMP^*
\alpha A t z,
\qquad
\alpha=\alpha^\prime\beta ,
\qquad
t^\prime z^\prime\in T^{\prime *},
\]
in which the last segment rewrites only the suffix,
\(\gamma t^\prime z^\prime\RMP^*tz\), leaving this \(A\) untouched.  Splitting on
whether \(\gamma\) is nullable makes exactly one alternative hold:
\[
\begin{cases}
\gamma=\gamma_0X\gamma_1,\quad
\gamma_0\Rightarrow_{P^\prime}^*\eps,\quad
X\RMP^*t\chi,
&
t\in\mathsf{Read}(p,A),\\[2pt]
\gamma\Rightarrow_{P^\prime}^*\eps,\quad t=t^\prime,
&
t\in\mathsf{Follow}(p^\prime,B)
\text{ and }
(p,A)\leadsto_{\mathrm{includes}}(p^\prime,B).
\end{cases}
\]
In the first case \(\gamma_0\) is read off \(\delta(p,A)\) by a nullable
\(\delta\)-chain and \(t\in\mathrm{FIRST}(X)\) gives \(\delta(\cdot,t)\ne\emptyset\),
so \(\gamma_0t:\delta(p,A)\to s\) realizes \eqref{eq:sem-read}.  In the second,
\(\gamma\Rightarrow_{P^\prime}^*\eps\) forces \(t=t^\prime\), and
\[
S^\prime\RMP^*\alpha^\prime B t z,\quad
\Path{\alpha^\prime}{q_0}{p^\prime}\ (\text{\eqref{eq:path-concat} on }\alpha=\alpha^\prime\beta)
\ \Longrightarrow\
t\in\mathsf{Follow}(p^\prime,B),
\]
the includes edge holding by \(\Item{B}{\beta\cdot A\gamma}\in p\),
\(\Path{\beta}{p^\prime}{p}\), \(\gamma\Rightarrow_{P^\prime}^*\eps\).  Its
witnessing derivation \(S^\prime\RMP^*\alpha^\prime Btz\) is strictly shorter, so
induction on the length gives
\[
\mathsf{Follow}(p,A)
\subseteq
\mu\Phi_F(p,A).
\]
Together with closure, this proves \eqref{eq:follow-eq}.
\end{proof}

\begin{corollary}[Lookahead by Follow]\label{cor:lookahead-follow}
Whenever \(q\in Q\) and \(\Item{A}{\omega\cdot\eps}\in q\),
\begin{equation}\label{eq:la-follow-formula}
\Look(q,\Item{A}{\omega})
=
\set*{t\in T^\prime
\mid
\exists p\in Q.\ 
\Path{\omega}{p}{q}
\land
\delta(p,A)\ne\emptyset
\land
t\in\Follow\mathbin{!}(p,A)}.
\end{equation}
\end{corollary}

\begin{proof}
For \(q\in Q\) and \(\Item{A}{\omega\cdot\eps}\in q\),
\[
\begin{aligned}
t\in\Look(q,\Item{A}{\omega})
&\Longleftrightarrow
\exists\alpha,z.\
S^\prime\RMP^*\alpha A t z
\land
\Path{\alpha\omega}{q_0}{q}\\
&\overset{\eqref{eq:path-concat}}{\Longleftrightarrow}
\exists p,\alpha,z.\
S^\prime\RMP^*\alpha A t z
\land
\Path{\alpha}{q_0}{p}
\land
\Path{\omega}{p}{q}\\
&\overset{\eqref{eq:sem-follow}}{\Longleftrightarrow}
\exists p.\
\Path{\omega}{p}{q}
\land
\delta(p,A)\ne\emptyset
\land
t\in\mathsf{Follow}(p,A)\\
&\overset{\eqref{eq:follow-eq}}{\Longleftrightarrow}
\exists p.\ 
\Path{\omega}{p}{q}
\land
\delta(p,A)\ne\emptyset
\land
t\in\Follow\mathbin{!}(p,A).
\end{aligned}
\]
The side condition \(\delta(p,A)\ne\emptyset\) is automatic for \(A\ne S^\prime\):
tracing \(\Item{A}{\omega\cdot\eps}\in q\) backward along
\(\Path{\omega}{p}{q}\) gives \(\Item{A}{\eps\cdot\omega}\in p\), which is
introduced by an item with the dot before \(A\).  For \(A=S^\prime\), both sides
are empty because \(S^\prime\) occurs on no right-hand side and the parser does
not reduce \(\Item{S^\prime}{S\$}\).  This is \eqref{eq:la-follow-formula}.
\end{proof}

\begin{theorem}[Digraph computation of lookahead]\label{thm:read-follow}
The least relations \(\Read,\Follow\) generated above compute every lookahead
set by the formula \eqref{eq:la-follow-formula}.
\end{theorem}

\begin{proof}
The Read and Follow fixed-point claims are Lemmas~\ref{lem:read-fixed-point}
and~\ref{lem:follow-fixed-point}; applying them to the semantic definition of
\(\Look\) gives Corollary~\ref{cor:lookahead-follow}, which is the displayed
formula of the theorem.
\end{proof}

\section{Lookahead Restriction}

Recall the lookahead set \(\Look\) of \eqref{eq:la-def}.  Replace the LR(0) reduction function by
\begin{equation}\label{eq:la-reduce}
\Reduce_{\Look}(q,t)
\triangleq
\set*{\Item{A}{\omega}
\mid
\Item{A}{\omega\cdot\eps}\in q,\
t\in\Look(q,\Item{A}{\omega})}.
\end{equation}

Let \(\vdash_{\Look}\) be the transition relation obtained from
\(\vdash\) by replacing \(\Reduce\) with \(\Reduce_{\Look}\), and let
\(L(\vdash_{\Look})\) be its accepted language.

\begin{lemma}[Guarded soundness]\label{lem:lookahead-sound}
\[
L(\vdash_{\Look})\subseteq L(G).
\]
\end{lemma}

\begin{proof}
From \eqref{eq:la-reduce},
\[
\Reduce_{\Look}(q,t)\subseteq\Reduce(q,t)
\quad\Longrightarrow\quad
\vdash_{\Look}\subseteq\vdash
\quad\Longrightarrow\quad
\vdash_{\Look}^{\,*}\subseteq\vdash^*.
\]
Therefore
\begin{equation}\label{eq:la-sound}
L(\vdash_{\Look})
\subseteq
L(\LRA(G))
=
L(G)
\qquad\text{by coro~\ref{coro:lra-correct}.}
\end{equation}
\end{proof}

\begin{lemma}[Guarded completeness]\label{lem:lookahead-complete}
\[
L(G)\subseteq L(\vdash_{\Look}).
\]
\end{lemma}

\begin{proof}
Let \(z\in L(G)\).  In the completeness construction of
coro~\ref{coro:lra-correct}, every Reduce step has the form
\[
(\Path{\alpha\omega}{q_0}{q},t z^\prime)
\vdash
(\Path{\alpha A}{q_0}{\delta(p,A)},t z^\prime),
\]
where
\[
S^\prime\RMP S\$\RMP^*\alpha A t z^\prime,
\qquad
\Path{\alpha\omega}{q_0}{q},
\qquad
\Item{A}{\omega\cdot\eps}\in q.
\]
By \eqref{eq:la-def},
\[
t\in\Look(q,\Item{A}{\omega})
\quad\Longleftrightarrow\quad
\Item{A}{\omega}\in\Reduce_{\Look}(q,t).
\]
Thus every Reduce step used in that accepting LR(0) computation is also legal
for \(\vdash_{\Look}\); Shift steps are unchanged.  Hence
\[
L(G)\subseteq L(\vdash_{\Look}).
\]
\end{proof}

\begin{corollary}[Language preservation]\label{cor:lookahead-language}
\[
L(\vdash_{\Look})=L(G).
\]
\end{corollary}

\begin{proof}
Combine Lemmas~\ref{lem:lookahead-sound} and~\ref{lem:lookahead-complete}.
\end{proof}

\begin{lemma}[Conflict-free table condition]\label{lem:lookahead-table}
If the resulting table has no shift/reduce or reduce/reduce conflict, its
action is single-valued under one-symbol lookahead, hence it is an
\(\LALRone\) table.
\end{lemma}

\begin{proof}
The action at \((q,t)\) consists of the shift
\(\delta(q,t)\) when it is nonempty and the reductions in
\(\Reduce_{\Look}(q,t)\).  Absence of shift/reduce and reduce/reduce conflicts
means that this action is single-valued under one-symbol lookahead, which is
precisely the \(\LALRone\) table condition.
\end{proof}

\begin{theorem}[Lookahead-guarded reductions]\label{thm:lookahead-guard}
\leavevmode\par\noindent
If the resulting table has no shift/reduce or reduce/reduce conflict, it is an
\(\LALRone\) parser and accepts \(L(G)\).
\end{theorem}

\begin{proof}
Corollary~\ref{cor:lookahead-language} gives acceptance of \(L(G)\), and
Lemma~\ref{lem:lookahead-table} gives the \(\LALRone\) table condition under the
stated absence of conflicts.
\end{proof}

\appendix
\section{Digraph Algorithm and Implementation Specification}

This appendix gives an executable specification for computing the sets used in
Theorem~\ref{thm:read-follow}.  The construction is the usual digraph view of
\(\LALRone\) lookahead computation: direct seed sets are placed on nodes,
dependency edges describe set inclusions, and the least fixed point is obtained
by collapsing strongly connected components and propagating set unions along the
resulting DAG
\cite{DeRemerPennello1982,Tarjan1972,Gallier2008}.

\subsection{General Digraph Fixed-Point Problem}

Let \(X\) be a finite set of nodes, let \(E\subseteq X\times X\) be a directed
edge relation, and let \(\mathrm{Seed}:X\to\Pow(T^\prime)\).  We use the edge
orientation
\[
x\leadsto y
\qquad\text{to mean}\qquad
F(y)\subseteq F(x).
\]
Thus the desired solution is the least function \(F:X\to\Pow(T^\prime)\)
satisfying
\begin{equation}\label{eq:digraph}
F(x)=\mathrm{Seed}(x)\cup\bigcup_{x\leadsto y}F(y).
\end{equation}

\paragraph{Algorithm \(\mathrm{DIGRAPH}(X,E,\mathrm{Seed})\).}
\begin{enumerate}
\item Compute the strongly connected components of \((X,E)\), for example by
Tarjan's depth-first-search algorithm.
\item Build the condensation DAG \(\mathcal C\).  Its nodes are SCCs, and
\(C\leadsto D\) is an edge when some \(x\in C\) and \(y\in D\) satisfy
\(x\leadsto y\), with \(C\ne D\).
\item Initialize, for every SCC \(C\),
\[
\mathrm{Val}(C)\leftarrow
\bigcup_{x\in C}\mathrm{Seed}(x).
\]
\item Process the SCCs in reverse topological order with respect to
\(\leadsto\), so every successor \(D\) of \(C\) has already been processed, and
put
\[
\mathrm{Val}(C)
\leftarrow
\mathrm{Val}(C)\cup
\bigcup_{C\leadsto D}\mathrm{Val}(D).
\]
\item Return \(F(x)\triangleq\mathrm{Val}(C_x)\), where \(C_x\) is the SCC
containing \(x\).
\end{enumerate}

\paragraph{Correctness condition.}
For every SCC \(C\), after step 4,
\[
\mathrm{Val}(C)=
\bigcup
\set*{\mathrm{Seed}(y)
\mid
\text{some }x\in C\text{ reaches }y\text{ in }(X,E)}.
\]
Therefore the returned \(F\) is exactly the least solution of
\eqref{eq:digraph}.  This is the only property of the graph algorithm needed by
Theorem~\ref{thm:read-follow}.

\paragraph{Complexity.}
Let \(b=\lceil |T^\prime|/w\rceil\), where \(w\) is the machine word size used
for bitsets.  SCC construction is \(O(|X|+|E|)\), and set propagation is
\[
O\bigl((|X|+|E|)b\bigr).
\]
A simple work-list implementation that repeatedly applies
\[
F(x):=F(x)\cup F(y)\qquad(x\leadsto y)
\]
is also correct, but the SCC version avoids repeated propagation inside cycles.

\subsection{\texorpdfstring{Instance for \(\Read\)}{Instance for Read}}

First compute nullable nonterminals by the least fixed point
\[
\mathrm{Null}(A)
\Longleftrightarrow
\exists \Item{A}{X_1\cdots X_k}\in P^\prime
\text{ such that every }X_i\text{ is nullable},
\]
where terminals are never nullable and \(k=0\) is allowed.  Extend this to
strings by
\[
\mathrm{Null}(X_1\cdots X_k)
\Longleftrightarrow
\bigwedge_{i=1}^k \mathrm{Null}(X_i).
\]

For \(x=(p,A)\in D\), define the direct-read seed
\[
\mathrm{DR}(p,A)
\triangleq
\set*{t\in T^\prime\mid \delta(\delta(p,A),t)\ne\emptyset}.
\]
Define \(E_R\subseteq D\times D\) by
\[
(p,A)\leadsto_R(r,C)
\Longleftrightarrow
\delta(p,A)=r
\land C\in N^\prime
\land \mathrm{Null}(C)
\land (r,C)\in D.
\]
If \((r,C)\notin D\), then \(\Read\mathbin{!}(r,C)=\emptyset\), so no edge is
needed.  Compute
\[
\Read\mathbin{!}(p,A)
\triangleq
\mathrm{DIGRAPH}(D,E_R,\mathrm{DR})(p,A).
\]

\subsection{\texorpdfstring{Instance for \(\Follow\)}{Instance for Follow}}

The seed for follow propagation is the already computed read set:
\[
\mathrm{Seed}_F(p,A)\triangleq\Read\mathbin{!}(p,A).
\]
Define \(E_I\subseteq D\times D\) by
\[
(p,A)\leadsto_I(p^\prime,B)
\Longleftrightarrow
\Item{B}{\beta\cdot A\gamma}\in p
\land
\Path{\beta}{p^\prime}{p}
\land
\mathrm{Null}(\gamma)
\land
(p^\prime,B)\in D.
\]
The orientation means
\[
\Follow\mathbin{!}(p^\prime,B)\subseteq\Follow\mathbin{!}(p,A).
\]
Compute
\[
\Follow\mathbin{!}(p,A)
\triangleq
\mathrm{DIGRAPH}(D,E_I,\mathrm{Seed}_F)(p,A).
\]

\subsection{Lookback Relation and Lookahead Construction}

For every completed item \(\Item{A}{\omega\cdot\eps}\in q\), define
\[
\mathrm{LB}(q,A\to\omega)
\triangleq
\set*{(p,A)\in D\mid \Path{\omega}{p}{q}}.
\]
Then
\[
\Look(q,\Item{A}{\omega})
\triangleq
\bigcup_{(p,A)\in\mathrm{LB}(q,A\to\omega)}
\Follow\mathbin{!}(p,A),
\]
which is \eqref{eq:la-follow-formula} written as an implementation step.

\paragraph{Implementation note.}
The query \(\Path{\omega}{p}{q}\) can be implemented by following
\(\delta\)-edges from \(p\) along \(\omega\).  The query
\(\Path{\beta}{p^\prime}{p}\) in \(E_I\) can be implemented by caching
reverse-path queries keyed by \((\beta,p)\), or by storing predecessor edges
during construction of the LR(0) automaton.

\subsection{Parser-Table and Conflict Specification}

After \(\Look\) has been computed, construct actions as follows.
\begin{enumerate}
\item If \(\delta(q,t)\ne\emptyset\) for \(t\in T^\prime\), add the shift action
\[
q\xrightarrow{t}\delta(q,t).
\]
\item For each \(\Item{A}{\omega\cdot\eps}\in q\) and each
\(t\in\Look(q,\Item{A}{\omega})\), add the reduce action \(A\to\omega\) under
lookahead \(t\).
\item Report a shift/reduce conflict when both a shift action and a reduce
action are present for the same pair \((q,t)\).
\item Report a reduce/reduce conflict when two distinct reduce productions are
present for the same pair \((q,t)\).
\end{enumerate}
The accepting condition remains the one in Definition~\ref{def:lra}: the parser
accepts when it reaches
\[
(\Path{S\$}{q_0}{q_f},\eps).
\]
Equivalently, in a conventional ACTION/GOTO table, \(q_f\) may be treated as the
accepting state after the input marker has been consumed.

\subsection{Summary Specification}

The complete computation required by Theorem~\ref{thm:read-follow} is:
\begin{enumerate}
\item Build \(G^\prime\), \(Q\), and \(\delta\) as in
Definition~\ref{def:lra}.
\item Compute nullable nonterminals and nullable strings.
\item Build \(D\), \(\mathrm{DR}\), and \(E_R\); compute \(\Read\) by
\(\mathrm{DIGRAPH}(D,E_R,\mathrm{DR})\).
\item Build \(E_I\); compute \(\Follow\) by
\(\mathrm{DIGRAPH}(D,E_I,\Read)\).
\item Build \(\mathrm{LB}\) and compute each
\(\Look(q,\Item{A}{\omega})\) by unioning the appropriate
\(\Follow\)-sets.
\item Construct the guarded reduce actions and check conflicts.
\end{enumerate}
The postcondition is that the computed \(\Read\), \(\Follow\), and \(\Look\) are
the least sets satisfying Theorem~\ref{thm:read-follow}, and hence the
lookahead-restricted parser of Theorem~\ref{thm:lookahead-guard} uses exactly
the \(\LALRone\) lookahead sets determined by the LR(0) automaton.
