\begin{definition}[\textbf{LR(0)}]
Let $G \triangleq ( N , T , P , S )$ be a CFG.
\begin{enumerate} \setlength{\itemsep}{0.2pt} \setlength{\parskip}{0.2pt}
\item[(i)] $G^\prime \triangleq ( N^\prime , T^\prime , P^\prime , S^\prime ) = ( N \uplus \set*{ S^\prime } , T \uplus \set*{ \$ } , P \cup \set*{ [ S^\prime \to S ] } , S^\prime )$.
\item[(ii)] $V \triangleq N \uplus T$. $V^\prime \triangleq N^\prime \uplus T^\prime$.
\item[(iii)] $\alpha_1 \xRightarrow[\mathbf{rm}]{}_{P^\prime} \alpha_2 \xLeftrightarrow[]{\mathrm{def}} ( \alpha_1 , \alpha_2 ) \in \set*{ ( \alpha A z , \alpha \omega z ) \mid [ A \to \omega ] \in P^\prime , \alpha \in ( V^\prime )^\ast , z \in ( T^\prime )^\ast }$.
\item[(iv)] $L ( G ) \triangleq \set*{ z \in T^\ast \mid S \Rightarrow_P^\ast z }$. $L ( G^\prime ) \triangleq \set*{ z \in T^\ast \mid S^\prime \xRightarrow[\mathbf{rm}]{}_{P^\prime}^+ z }$. $L ( G ) = L ( G^\prime )$.
\item[(v)] $I_{\text{LR(0)}} \triangleq \set*{ [ A \to \alpha \cdot \beta ] \mid [ A \to \alpha \beta ] \in P^\prime }$.
\item[(vi)] $\mathrm{Cl} ( q ) =_\mu q \cup \set*{ [ A \to \cdot \omega ] \mid [ A \to \omega ] \in P^\prime , [ B \to \alpha_1 \cdot A \alpha_2 ] \in \mathrm{Cl} ( q ) }$ for $ q \in \powerset ( I_{\text{LR(0)}} ) $.
\item[(vii)] $q_0 := \mathrm{Cl} ( \set*{ [ S^\prime \to \cdot S ] } )$.
\item[(viii)] $\mathrm{GOTO} ( q , X ) := \mathrm{Cl} ( \set*{ [ A \to \alpha X \cdot \beta ] \mid [ A \to \alpha \cdot X \beta ] \in q } )$ for $ q \in \powerset ( I_{\text{LR(0)}} ) $ and $ X \in V $.
\item[(ix)] $Q \triangleq \mathrm{Table} \setminus \set*{ \emptyset }$ where $\mathrm{Table} =_\mu \set*{ q_0 } \cup \set*{ \mathrm{GOTO} ( q , X ) \mid q \in \mathrm{Table} , X \in V }$.
\item[(x)] $\varepsilon : p \to q \xLeftrightarrow[]{\mathrm{def}} p \in Q \land p = q$. $X \alpha : p \to q \xLeftrightarrow[]{\mathrm{def}} p \in Q \land \alpha : \mathrm{GOTO} ( p , X ) \to q$.
\item[(xi)] $\mathrm{next} ( q , X ) := \mathrm{GOTO} ( q , X )$ for $q \in Q$ and $X \in V$.
\item[(xii)] $\mathrm{reduce} ( q , t ) := \set*{ [ A \to \omega ] \mid [ A \to \omega \cdot ] \in q , t \in \mathbf{LA} ( q , [ A \to \omega ] ) }$ for $q \in Q$ and $t \in T^\prime$.
\item[(xiii)] $$\infer[\mathrm{Shift}]{ ( \alpha : q \to q^\prime , t z ) \vdash ( \alpha t : q \to q^{\prime \prime} , z ) }{ q^{\prime \prime} = \mathrm{next} ( q^\prime , t ) & q^{\prime \prime} \ne \emptyset }$$
\item[(xiv)] $$\infer[\mathrm{Reduce} (A \to \omega)]{ ( \alpha \omega : q \to q^\prime , t z ) \vdash ( \alpha A : q \to q^{\prime \prime} , t z ) }{ [ A \to \omega ] \in \mathrm{reduce} ( q^\prime , t ) & q^{\prime \prime} = \mathrm{next} ( q^\prime , A ) & q^{\prime \prime} \ne \emptyset }$$
\item[(xv)] $q_f := \mathrm{GOTO} ( q_0 , S )$.
\item[(xvi)] $L ( LRA ( G ) ) \triangleq \set*{ z \in T^\ast \mid ( \varepsilon : q_0 \to q_0 , z \$ ) \vdash^+ ( S : q_0 \to q_f , \$ ) }$. $L ( G ) = L ( LRA ( G ) )$.
\item[(xvii)] For $ts_1 \subseteq \set*{ z \in ( T^\prime )^\ast \mid \left| z \right| \le 1 }$ and $ts_2 \subseteq \set*{ z \in ( T^\prime )^\ast \mid \left| z \right| \le 1 }$, \[ ts_1 \odot ts_2 := \begin{cases} ( ts_1 \setminus \set*{ \varepsilon } ) \cup ts_2 , & \text{if } \varepsilon \in ts_1 ; \\ ts_1 , & \text{if } \varepsilon \notin ts_1 . \end{cases} \]
\item[(xviii)] $\mathrm{First} ( A ) := \set*{ t \mid t \in T^\prime , \omega \in V^\ast , A \Rightarrow_{P^\prime}^\ast t \omega } \cup \set*{ \varepsilon \mid A \Rightarrow_{P^\prime}^\ast \varepsilon }$ for $A \in N^\prime$.
\item[(xix)] $\mathrm{First} ( t ) := \set*{ t }$ for $t \in T^\prime$.
\item[(xx)] $\mathrm{First} ( \varepsilon ) := \set*{ \varepsilon }$. $\mathrm{First} ( X \alpha ) := \mathrm{First} ( X ) \odot \mathrm{First} ( \alpha )$ for $X \in V^\prime$ and $\alpha \in ( V^\prime )^\ast $.
\end{enumerate}
\end{definition}

\begin{theorem}[\textbf{LA(1)}]
For $q \in Q$, $A \in N^\prime$, and $\omega \in V^\ast$ with $ [ A \to \omega \cdot ] \in q $, let
\begin{align}
\mathbf{LA} ( q , [ A \to \omega ] ) \triangleq \set*{ t \in T^\prime \mid S^\prime \xRightarrow[\mathbf{rm}]{}_{P^\prime}^\ast \alpha A z , \alpha \omega : q_0 \to q , \alpha \in V^\ast , z \in T^\ast , t \in \mathrm{First} ( z \$ ) } .
\end{align}
Then, it can be calculated by
\begin{align}
\mathbf{LA} ( q , [ A \to \omega ] ) = \set*{ t \in T^\prime \mid t \in \mathbf{LA}^\prime ( q , [ A \to \omega \cdot ] ) }
\end{align}
for some $\mathbf{LA}^\prime : \set*{ ( q , [ A \to \alpha \cdot \beta ] ) \mid q \in Q , [ A \to \alpha \cdot \beta ] \in q } \to \powerset ( \set*{ z \in ( T^\prime )^\ast \mid \left| z \right| \le 1 } )$ with:
\begin{align}
\mathbf{LA}^\prime ( q_0 , [ S^\prime \to \cdot S ] ) & = \set*{ \$ } \label{eq:initial} ; \\
\mathbf{LA}^\prime ( q_f , [ S^\prime \to S \cdot ] ) & = \set*{ \$ } \label{eq:final} ; \\
\mathbf{LA}^\prime ( q , [ A \to \alpha \cdot \beta ] ) & = \bigcup_{\alpha : p \to q} \bigcup_{[ B \to \alpha_1 \cdot A \alpha_2 ] \in p} \mathrm{First} ( \alpha_2 ) \odot \mathbf{LA}^\prime ( p , [ B \to \alpha_1 \cdot A \alpha_2 ] ) \label{eq:step} 
\end{align}
if $ A \ne S^\prime $.
\end{theorem}

\begin{proof}
Satisfying $\eqref{eq:initial}$, $\eqref{eq:final}$, and $\eqref{eq:step}$, there is a map $\mathbf{LA}^\prime$ such that
\begin{align}
\mathbf{LA}^\prime ( q , [ A \to \alpha \cdot \beta ] ) = \bigcup \set*{ \mathrm{First} ( z \$ ) \mid S^\prime \xRightarrow[\mathbf{rm}]{}_{P^\prime}^\ast \omega A z \xRightarrow[\mathbf{rm}]{}_{P^\prime} \omega \alpha \beta z , \omega \alpha : q_0 \to q , z \in T^\ast }
\end{align}
whenever $q \in Q$ and $[ A \to \alpha \cdot \beta ] \in q$.
\end{proof}
