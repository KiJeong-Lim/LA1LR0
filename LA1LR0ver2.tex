잇창명님께 감사드립니다. 다음은 LALR 파서를 생성하는 데에 필요한 정의이다.

\begin{enumerate} \setlength{\itemsep}{0.2pt} \setlength{\parskip}{0.2pt}
\item[(i)] Let $G \triangleq ( N , T , P , S )$ be a CFG.
\item[(ii)] $G^\prime \triangleq ( N^\prime , T^\prime , P^\prime , S^\prime ) = ( N \uplus \set*{ S^\prime } , T \uplus \set*{ \$ } , P \uplus \set*{ [ S^\prime \to S ] } , S^\prime )$.
\item[(iii)] $V \triangleq N \uplus T$. $V^\prime \triangleq N^\prime \uplus T^\prime$.
\item[(iv)] $L ( G ) := \set*{ z \in T^\ast \mid S \Rightarrow_P^\ast z }$. $L ( G^\prime ) := \set*{ z \in T^\ast \mid S^\prime \xRightarrow[\mathbf{rm}]{}_{P^\prime}^+ z }$. $L ( G ) = L ( G^\prime )$.
\item[(v)] $I \triangleq \set*{ [ A \to \alpha \cdot \beta ] \mid [ A \to \alpha \beta ] \in P^\prime }$.
\item[(vi)] $\mathrm{Cl} ( q ) =_\mu q \cup \set*{ [ A \to \cdot \omega ] \mid [ A \to \omega ] \in P^\prime , [ B \to \alpha_1 \cdot A \alpha_2 ] \in \mathrm{Cl} ( q ) }$ for $ q \in \powerset ( I ) $.
\item[(vii)] $q_0 \triangleq \mathrm{Cl} ( \set*{ [ S^\prime \to \cdot S ] } )$.
\item[(viii)] $\mathrm{GOTO} ( q , X ) \triangleq \mathrm{Cl} ( \set*{ [ A \to \alpha X \cdot \beta ] \mid [ A \to \alpha \cdot X \beta ] \in q } )$ for $ q \in \powerset ( I ) $ and $ X \in V $.
\item[(ix)] $Q \triangleq \mathrm{PT} \setminus \set*{ \emptyset }$ where $\mathrm{PT} =_\mu \set*{ q_0 } \cup \set*{ \mathrm{GOTO} ( q , X ) \mid q \in \mathrm{PT} , X \in V }$.
\item[(x)] $\varepsilon : p \to q \xLeftrightarrow[]{\mathrm{def}} p \in Q \land p = q$. $X \alpha : p \to q \xLeftrightarrow[]{\mathrm{def}} p \in Q \land \alpha : \mathrm{GOTO} ( p , X ) \to q$.
\item[(xi)] $\mathrm{next} ( q , X ) := \mathrm{GOTO} ( q , X )$ for $q \in Q$ and $X \in V$.
\item[(xii)] $\mathrm{reduce} ( q , t ) := \set*{ [ A \to \omega ] \mid [ A \to \omega \cdot ] \in q , t \in \mathbf{LA} ( q , [ A \to \omega ] ) }$ for $q \in Q$ and $t \in T^\prime$.
\item[(xiii)] $$\infer[\mathrm{Shift}]{ ( \alpha : q \to q^\prime , t z ) \vdash ( \alpha t : q \to q^{\prime \prime} , z ) }{ q^{\prime \prime} = \mathrm{next} ( q^\prime , t ) & q^{\prime \prime} \ne \emptyset }$$
\item[(xiv)] $$\infer[\mathrm{Reduce} (A \to \omega)]{ ( \alpha \omega : q \to q^\prime , t z ) \vdash ( \alpha A : q \to q^{\prime \prime} , t z ) }{ [ A \to \omega ] \in \mathrm{reduce} ( q^\prime , t ) & q^{\prime \prime} = \mathrm{next} ( q^\prime , A ) & q^{\prime \prime} \ne \emptyset }$$
\item[(xv)] $q_f := \mathrm{GOTO} ( q_0 , S )$.
\item[(xvi)] $L ( LRA ( G ) ) := \set*{ z \in T^\ast \mid ( \varepsilon : q_0 \to q_0 , z \$ ) \vdash^+ ( S : q_0 \to q_f , \$ ) }$. $L ( G ) = L ( LRA ( G ) )$.
\item[(xvii)] For $ts_1 \subseteq \set*{ z \in ( T^\prime )^\ast \mid \left| z \right| \le 1 }$ and $ts_2 \subseteq \set*{ z \in ( T^\prime )^\ast \mid \left| z \right| \le 1 }$, \[ ts_1 \odot ts_2 := \begin{cases} ts_1 , & \mathrm{if} \, \varepsilon \notin ts_1 ; \\ ( ts_1 \setminus \set*{ \varepsilon } ) \cup ts_2 , & \mathrm{otherwise} . \end{cases} \]
\item[(xviii)] $\mathrm{First} ( A ) := \set*{ t \mid t \in T^\prime , \omega \in V^\ast , A \Rightarrow_{P^\prime}^\ast t \omega } \cup \set*{ \varepsilon \mid A \Rightarrow_{P^\prime}^\ast \varepsilon }$ for $A \in N^\prime$.
\item[(xix)] $\mathrm{First} ( t ) := \set*{ t }$ for $t \in T^\prime$.
\item[(xx)] $\mathrm{First} ( \varepsilon ) := \set*{ \varepsilon }$. $\mathrm{First} ( X \alpha ) := \mathrm{First} ( X ) \odot \mathrm{First} ( \alpha )$ for $X \in V^\prime$ and $\alpha \in ( V^\prime )^\ast $.
\end{enumerate}

\noindent \textbf{Theorem. (LA)}
$ [ A \to \omega \cdot ] \in q $인 $q \in Q$, $A \in N^\prime$와 $\omega \in V^\ast$에 대하여,
\begin{align}
\mathbf{LA} ( q , [ A \to \omega ] ) \triangleq \set*{ t \in T^\prime \mid \begin{matrix} (\alpha \omega : q_0 \to q , t z) \vdash_{\mathrm{Reduce} ( A \to \omega )} ( \alpha A : q_0 \to q^\prime , t z ) \\ \andb \\ ( \alpha A : q_0 \to q^\prime , t z ) \vdash^\ast ( S : q_0 \to q_f , \$ ) \end{matrix} } .
\end{align}
라 하자. 그러면
\begin{align}
\mathbf{LA} ( q , [ A \to \omega ] ) = \set*{ t \in T^\prime \mid S^\prime \xRightarrow[\mathbf{rm}]{}_{P^\prime}^\ast \alpha A z , \alpha \omega : q_0 \to q , \alpha \in V^\ast , z \in T^\ast , t \in \mathrm{First} ( z \$ ) }
\end{align}
이 성립한다.

\noindent 이제 $\mathbf{LA}^\prime : \set*{ ( q , [ A \to \alpha \cdot \beta ] ) \mid q \in Q , [ A \to \alpha \cdot \beta ] \in q } \to \powerset ( \set*{ z \in ( T^\prime )^\ast \mid \left| z \right| \le 1 } )$를
\begin{align}
\mathbf{LA}^\prime =_\mu \mathbf{propagate}_{\mathbf{LA}^\prime} ,
\end{align}
로 정의하자. 여기서 $\mathbf{propagate}$의 도입 규칙은 모두 3가지로
\begin{equation}
\begin{matrix} \\ \infer[\mathbf{Initial}]{\$ \in \mathbf{propagate}_{\mathbf{LA}^\prime_\mathrm{previous}} ( q_0 , [ S^\prime \to \cdot S ] )}{}\end{matrix}
\end{equation}
\begin{equation}
\begin{matrix} \\ \infer[\mathbf{Final}]{\$ \in \mathbf{propagate}_{\mathbf{LA}^\prime_\mathrm{previous}} ( q_f , [ S^\prime \to S \cdot ] )}{}\end{matrix}
\end{equation}
\begin{equation}
\begin{matrix}\infer[\mathbf{Step}]{\mathrm{First} ( \alpha_2 ) \odot \mathbf{LA}^\prime_\mathrm{previous} ( p , [ B \to \alpha_1 \cdot A \alpha_2 ] ) \subseteq \mathbf{propagate}_{\mathbf{LA}^\prime_\mathrm{previous}} ( q , [ A \to \alpha \cdot \beta ] )}{q \in Q & [ A \to \alpha \cdot \beta ] \in q & [ A \to \alpha \beta ] \in P & \alpha : p \to q & [ B \to \alpha_1 \cdot A \alpha_2] \in p}\end{matrix}
\end{equation}
% \begin{equation}
% \begin{matrix}\infer[\mathbf{Extensive}]{\mathbf{LA}^\prime_\mathrm{previous} ( q , [ A \to \alpha \cdot \beta ] ) \subseteq \mathbf{propagate}_{\mathbf{LA}^\prime_\mathrm{previous}} ( q , [ A \to \alpha \cdot \beta ] )}{q \in Q & [ A \to \alpha \cdot \beta ] \in q}\end{matrix}
% \end{equation}
\noindent 이다. 그러면, $q \in Q$이고 $ [ A \to \omega \cdot ] \in q $일 때마다,
\begin{align}
\mathbf{LA} ( q , [ A \to \omega ] ) = \set*{ t \in T^\prime \mid t \in \mathbf{LA}^\prime ( q , [ A \to \omega \cdot ] ) }
\end{align}
\noindent 이 성립한다. 이는
\begin{align}
\mathbf{LA}^\prime ( q , [ A \to \alpha \cdot \beta ] ) = \bigcup \set*{ \mathrm{First} ( z \$ ) \mid S^\prime \xRightarrow[\mathbf{rm}]{}_{P^\prime}^\ast \omega A z \xRightarrow[\mathbf{rm}]{}_{P^\prime} \omega \alpha \beta z , \omega \alpha : q_0 \to q , z \in T^\ast }
\end{align}
\noindent 이기 때문이다.

\noindent \textbf{Remark. (Fixed-point)} 완비격자 $( D , \le )$가 주어졌다고 하자. 함수 $$f : D \to D$$가 단조증가할 때, $x_0 \in D$ 이상의 $f$의 고정점 중 가장 작은 것을 구하는 법을 생각해보자.
\begin{align*}
x : \mathsf{Ord} & \to D \\
S ( \alpha ) & \mapsto x_0 \vee f ( x_\alpha ) \\
\alpha & \mapsto \bigvee_{\zeta < \alpha} x_\zeta \quad \mathrm{if} \, \alpha \in \mathsf{LimOrd}_{>0}
\end{align*}
로 잡고,
$\kappa := \mathbf{h} ( D )$로 놓으면,
$ x_\kappa $가 구하는 그것이 된다.
즉,
\begin{align}
x_\kappa = f ( x_\kappa )
\end{align}
이고
\begin{align}
( \forall x \in D ) ( x = f ( x ) \to ( x_0 \le x \leftrightarrow x_\kappa \le x ) )
\end{align}
이다.
여기서, 임의의 집합 $A$에 대하여 \[\mathbf{h} ( A ) \triangleq \set*{ \alpha \in \mathsf{Ord} \mid \exists \, \iota : \alpha \hookrightarrow A } \] 이며,
$A$가 유한하면 $\mathbf{h} ( A ) = \left| A \right| + 1$이다.
