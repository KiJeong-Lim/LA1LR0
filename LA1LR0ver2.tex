\begin{definition}[\textbf{LR(0)}]
Let $G \triangleq ( N , T , P , S )$ be a CFG.
\begin{enumerate}
\item[(i)] $G^\prime \triangleq ( N^\prime , T^\prime , P^\prime , S^\prime ) = ( N \uplus \set*{ S^\prime } , T \uplus \set*{ \$ } , P \cup \set*{ [ S^\prime \to S ] } , S^\prime )$.
\item[(ii)] $V \triangleq N \uplus T$. $V^\prime \triangleq N^\prime \uplus T^\prime$.
\item[(iii)] $\alpha_1 \xRightarrow[\mathbf{rm}]{}_{P^\prime} \alpha_2 \xLeftrightarrow[]{\mathrm{def}} ( \alpha_1 , \alpha_2 ) \in \set*{ ( \alpha A z , \alpha \omega z ) \mid [ A \to \omega ] \in P^\prime , \alpha \in ( V^\prime )^\ast , z \in ( T^\prime )^\ast }$.
\item[(iv)] $L ( G ) \triangleq \set*{ z \in T^\ast \mid S \Rightarrow_\mathrm{prev}^\ast z }$. $L ( G^\prime ) \triangleq \set*{ z \in ( T^\prime )^\ast \mid S^\prime \xRightarrow[\mathbf{rm}]{}_{P^\prime}^+ z }$. $L ( G ) = L ( G^\prime )$.
\item[(v)] $I_{\text{LR(0)}} \triangleq \set*{ [ A \to \alpha \cdot \beta ] \mid [ A \to \alpha \beta ] \in P^\prime }$.
\item[(vi)] $\mathrm{Cl} ( q ) =_\mu q \cup \set*{ [ A \to \cdot \omega ] \mid [ A \to \omega ] \in P^\prime , [ B \to \alpha_1 \cdot A \alpha_2 ] \in \mathrm{Cl} ( q ) }$ for $ q \in \powerset ( I_{\text{LR(0)}} ) $.
\item[(vii)] $q_0 := \mathrm{Cl} ( \set*{ [ S^\prime \to \cdot S ] } )$.
\item[(viii)] $\mathrm{GOTO} ( q , X ) := \mathrm{Cl} ( \set*{ [ A \to \alpha X \cdot \beta ] \mid [ A \to \alpha \cdot X \beta ] \in q } )$ for $ q \in \powerset ( I_{\text{LR(0)}} ) $ and $ X \in V $.
\item[(ix)] $Q \triangleq \mathrm{Tbl} \setminus \set*{ \emptyset }$ where $\mathrm{Tbl} =_\mu \set*{ q_0 } \cup \set*{ \mathrm{GOTO} ( q , X ) \mid q \in \mathrm{Tbl} , X \in V }$.
\item[(x)] $\varepsilon : p \to q \xLeftrightarrow[]{\mathrm{def}} p \in Q \land p = q$. $X \alpha : p \to q \xLeftrightarrow[]{\mathrm{def}} p \in Q \land \alpha : \mathrm{GOTO} ( p , X ) \to q$.
\item[(xi)] $\mathrm{next} ( q , X ) := \mathrm{GOTO} ( q , X )$ for $q \in Q$ and $X \in V$.
\item[(xii)] $\mathrm{reduce} ( q , t ) := \set*{ [ A \to \omega ] \mid [ A \to \omega \cdot ] \in q , t \in \mathbf{LA} ( q , [ A \to \omega ] ) }$ for $q \in Q$ and $t \in T^\prime$.
\item[(xiii)] $$\infer[\mathrm{Shift}]{ ( \alpha : q \to q^\prime , t z ) \vdash ( \alpha t : q \to q^{\prime \prime} , z ) }{ q^{\prime \prime} = \mathrm{next} ( q^\prime , t ) & q^{\prime \prime} \ne \emptyset }$$
\item[(xiv)] $$\infer[\mathrm{Reduce} (A \to \omega)]{ ( \alpha \omega : q \to q^\prime , t z ) \vdash ( \alpha A : q \to q^{\prime \prime} , t z ) }{ [ A \to \omega ] \in \mathrm{reduce} ( q^\prime , t ) & q^{\prime \prime} = \mathrm{next} ( q^\prime , A ) & q^{\prime \prime} \ne \emptyset }$$
\item[(xv)] $q_f := \mathrm{GOTO} ( q_0 , S )$.
\item[(xvi)] $L ( LRA ( G ) ) \triangleq \set*{ z \in T^\ast \mid ( \varepsilon : q_0 \to q_0 , z \$ ) \vdash^+ ( S : q_0 \to q_f , \$ ) }$. $L ( G ) = L ( LRA ( G ) )$.
\item[(xvii)] For $ts_1 \subseteq \set*{ z \in ( T^\prime )^\ast \mid \left| z \right| \le 1 }$ and $ts_2 \subseteq \set*{ z \in ( T^\prime )^\ast \mid \left| z \right| \le 1 }$, \[ ts_1 \odot ts_2 := \begin{cases} ( ts_1 \setminus \set*{ \varepsilon } ) \cup ts_2 , & \text{if } \varepsilon \in ts_1 ; \\ ts_1 , & \text{if } \varepsilon \notin ts_1 . \end{cases} \]
\item[(xviii)] $\mathrm{First} ( A ) := \set*{ t \mid t \in T^\prime , \omega \in ( V^\prime )^\ast , A \Rightarrow_{P^\prime}^\ast t \omega } \cup \set*{ \varepsilon \mid A \Rightarrow_{P^\prime}^\ast \varepsilon }$ for $A \in N^\prime$.
\item[(xix)] $\mathrm{First} ( t ) := \set*{ t }$ for $t \in T^\prime$.
\item[(xx)] $\mathrm{First} ( \varepsilon ) := \set*{ \varepsilon }$. $\mathrm{First} ( X \alpha ) := \mathrm{First} ( X ) \odot \mathrm{First} ( \alpha )$ for $X \in V^\prime$ and $\alpha \in ( V^\prime )^\ast $.
\end{enumerate}
\end{definition}

\begin{theorem}[\textbf{LA(1)}]
$ [ A \to \omega \cdot ] \in q $인 $q \in Q$, $A \in N^\prime$와 $\omega \in ( V^\prime )^\ast$에 대하여,
\begin{align}
\mathbf{LA} ( q , [ A \to \omega ] ) \triangleq \set*{ t \in T^\prime \mid S^\prime \xRightarrow[\mathbf{rm}]{}_{P^\prime}^\ast \alpha A z , \alpha \omega : q_0 \to q , \alpha \in V^\ast , z \in T^\ast , t \in \mathrm{First} ( z \$ ) }
\end{align}
로 두자. 이제 $\mathbf{LA}^\prime \subseteq \set*{ ( ( q , [ A \to \alpha \cdot \beta ] ) , z ) \mid q \in Q , [ A \to \alpha \cdot \beta ] \in q , z \in ( T^\prime )^\ast , \left| z \right| \le 1 } $를
\begin{align*}
\mathbf{LA}^\prime =_\mu \textbf{Propagate}_{\mathbf{LA}^\prime}
\end{align*}
로 정의하자.
그러면, $q \in Q \land [ A \to \omega \cdot ] \in q$일 때마다
\begin{align}
\mathbf{LA} ( q , [ A \to \omega ] ) = \set*{ t \in T^\prime \mid ( ( q , [ A \to \omega \cdot ] ) , t ) \in \mathbf{LA}^\prime }
\end{align}
이 성립한다.
여기서 $\textbf{Propagate}$의 도입규칙은 모두 3가지로, $$M \mathbin{!} k := \set*{ v \mid ( k , v ) \in M }$$라고 하였을 때, 다음과 같은 것들이다:
\begin{equation}
\begin{matrix}\\ \infer[]{\$ \in \textbf{Propagate}_{\mathbf{LA}^\prime_\mathrm{prev}} \mathbin{!} ( q_0 , [ S^\prime \to \cdot S ] )}{}\end{matrix}
\end{equation}
\begin{equation}
\begin{matrix}\\ \infer[]{\$ \in \textbf{Propagate}_{\mathbf{LA}^\prime_\mathrm{prev}} \mathbin{!} ( q_f , [ S^\prime \to S \cdot ] )}{}\end{matrix}
\end{equation}
\begin{equation}
\begin{matrix}\\ \infer[]{\mathrm{First} ( \alpha_2 ) \odot \mathbf{LA}^\prime_\mathrm{prev} \mathbin{!} ( p , [ B \to \alpha_1 \cdot A \alpha_2 ] ) \subseteq \textbf{Propagate}_{\mathbf{LA}^\prime_\mathrm{prev}} \mathbin{!} ( q , [ A \to \alpha \cdot \beta ])}{q \in Q & [ A \to \alpha \cdot \beta ] \in q & A \ne S^\prime & \alpha : p \to q & [ B \to \alpha_1 \cdot A \alpha_2] \in p}\end{matrix}
\end{equation}
\end{theorem}
\begin{proof}
$q \in Q \land [ A \to \alpha \cdot \beta ] \in q$일 때마다
\begin{align}
\mathbf{LA}^\prime \mathbin{!} ( q , [ A \to \alpha \cdot \beta ] ) = \bigcup \set*{ \mathrm{First} ( z \$ ) \mid S^\prime \xRightarrow[\mathbf{rm}]{}_{P^\prime}^\ast \omega A z \xRightarrow[\mathbf{rm}]{}_{P^\prime} \omega \alpha \beta z , \omega \alpha : q_0 \to q , z \in T^\ast }
\end{align}
이기 때문이다.
\end{proof}

getLA :: Bool -> (ParserS, LR0Item) ->
  StateT (Map.Map (ParserS, LR0Item) (Bool, TerminalSet)) Identity TerminalSet
getLA final (q, item)
  | getLHS item == start' = return (TerminalSet (Set.singleton (Just TSEOF)))
  | otherwise = do
    m <- get
    case Map.lookup (q, item) m of
      Just (correct, TerminalSet ts)
        | correct || not final -> return (TerminalSet ts)
      _ -> do
        put (Map.insert (q, item) (False, TerminalSet Set.empty) m)
        result <- fmap (TerminalSet . Set.unions) dollar sequence
          [ fmap Set.unions dollar sequence
            [ do
              let ts = unTerminalSet (getFirstOf (tail (getRIGHT item')))
              if Nothing `Set.member` ts then do
                result <- getLA False (q', item')
                return ((Nothing `Set.delete` ts) `Set.union` unTerminalSet result)
              else return ts
            | item' <- Set.toList items'
            , getMarkSym item' == Just (NS (getLHS item))
            ]
          | (items', q') <- Map.toList (getVertices getCannonical0)
          , calcGOTO q' (getLEFT item) == Just q
          ]
        modify (Map.update (const (Just (True, result))) (q, item))
        return result
makeLATable :: Identity [((ParserS, ProductionRule), Set.Set TSym)]
makeLATable = do
  (triples, _) <- flip runStateT Map.empty dollar sequence
    [ do
      result <- getLA True (q, item)
      return
        ( (q, (getLHS item, getLEFT item))
        , Set.fromList [ t | Just t <- Set.toList (unTerminalSet result) ]
        )
    | (items, q) <- Map.toList (getVertices getCannonical0)
    , item <- Set.toList items
    , isNothing (getMarkSym item)
    ]
  return triples
